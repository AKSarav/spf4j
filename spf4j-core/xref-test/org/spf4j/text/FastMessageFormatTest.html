16    @Test
17    public void testFormatter() throws IOException {
18      StringBuilder sb = new StringBuilder();
19      MessageFormat format = new MessageFormat("{0}, {1}");
20      format.format(new Object [] {"a", "b"}, sb, null);
21      Assert.assertEquals("a, b", sb.toString());
22    }
