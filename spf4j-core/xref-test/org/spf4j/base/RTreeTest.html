5   package org.spf4j.base;

7   import org.spf4j.ds.RTree;
8   import java.util.List;
9   import junit.framework.Assert;
10  import org.junit.Test;
11  
12  /**
13   *
14   * @author zoly
15   */
16  public final class RTreeTest {
17      
18      /**
19       * Test of search method, of class RTree.
20       */
21      @Test
22      public void testSearch() {
23          RTree<String> rectangles = new RTree<String>();
24          rectangles.insert(new float [] {1, 1}, "Point 1,1");
25          rectangles.insert(new float [] {0.5f, 0.5f}, new float [] {1, 1}, "Rectangle 0.5, 0.5, 1, 1");
26          List<String> result = rectangles.search(new float [] {0, 0}, new float [] {2, 2});
27          Assert.assertEquals(2, result.size());
28          List<String> result2 = rectangles.search(new float [] {1.5f, 1.5f}, new float [] {2, 2});
29          Assert.assertEquals(1, result2.size());
30          
31          RTree<String> ranges = new RTree<String>(1);
32          ranges.insert(new float [] {1}, new float [] {1}, "range 1,2");
33          ranges.insert(new float [] {2}, new float [] {2}, "range 2,4");
34          Assert.assertEquals(0, ranges.search(new float [] {10}, new float [] {0}).size());
35      }
