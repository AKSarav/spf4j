15  /**
16   * test to validate the behavior between spf4j and FJP implementations...

19  public class ThreadPoolExecutorTest {
20  
21  
22    @Test(timeout = 10000)
23    public void testInteruptionBehavior() throws InterruptedException, ExecutionException {
24      LifoThreadPoolExecutorSQP executor = new LifoThreadPoolExecutorSQP("test", 0, 16, 60000, 0, 1024);
25      testPoolTaskCancellation(executor);
26    }
27  
28    @Test(timeout = 10000)
29    public void testInteruptionBehaviorFJP() throws InterruptedException, ExecutionException {
30      boolean assertError = false;
31      try {
32        testPoolTaskCancellation(new ForkJoinPool(16));
33      } catch (AssertionError err) {
34        System.err.println("expected " + err);
35        assertError = true;
36      }
37      Assert.assertTrue("expected that FJP tasks cannot be interrupted", assertError);
38    }
39  
40    public void testPoolTaskCancellation(ExecutorService executor) throws InterruptedException, ExecutionException {
41      RunnableImpl testRunnable = new RunnableImpl();
42      Future<?> submit = executor.submit(testRunnable);
43      Assert.assertTrue("task did not start", testRunnable.getStartedlatch().await(5, TimeUnit.SECONDS));
44      submit.cancel(true);
45      try {
46        submit.get();
47        Assert.fail("expected CancellationException");
48      } catch (CancellationException ex) {
49        // expected
50      }
51      Assert.assertTrue("task was not interrupted", testRunnable.getInterruptedLatch().await(5, TimeUnit.SECONDS));
52      executor.shutdown();
53      Assert.assertTrue("executor was not shut down", executor.awaitTermination(1000, TimeUnit.MILLISECONDS));
54    }
55  
56    static class RunnableImpl implements Runnable {
57  
58      private CountDownLatch startedlatch = new CountDownLatch(1);
59  
60      private CountDownLatch interruptedLatch = new CountDownLatch(1);
61  
62      @Override
63      @SuppressFBWarnings("MDM_THREAD_YIELD")
64      public void run() {
65        startedlatch.countDown();
66        try {
67          Thread.sleep(100000);
68        } catch (InterruptedException ex) {
69          interruptedLatch.countDown();
70        }
71      }
72  
73      public CountDownLatch getInterruptedLatch() {
74        return interruptedLatch;
75      }
76  
77  
78  
79  
80      public CountDownLatch getStartedlatch() {
81        return startedlatch;
82      }

87    }
