18  import com.google.common.base.Charsets;
19  import java.io.File;
20  import java.io.IOException;
21  import java.nio.file.Files;
22  import java.nio.file.Path;
23  import java.nio.file.Paths;
24  import java.util.stream.Collectors;
25  import org.hamcrest.Matchers;
26  import org.junit.Assert;
27  import org.junit.Test;
28  import org.spf4j.perf.impl.MeasurementsInfoImpl;
29  
30  /**
31   * @author zoly
32   */
33  public class TSDBTxtMeasurementStoreTest {
34  
35  
36    @Test
37    public void testCreateAppend() throws IOException {
38      Path tmpFile = Paths.get(org.spf4j.base.Runtime.TMP_FOLDER, "testM.txt");
39      File file = tmpFile.toFile();
40      TSDBTxtMeasurementStore store = new TSDBTxtMeasurementStore(file);
41      long id = store.alocateMeasurements(new MeasurementsInfoImpl("test", "bla",
42              new String[] {"a", "b"}, new String [] {"ms", "ms"}), 10000);
43      store.saveMeasurements(id, System.currentTimeMillis(), 3L, 4L);
44      store.close();
45      String content = Files.lines(tmpFile, Charsets.UTF_8).collect(Collectors.joining("\n"));
46      System.out.println(content);
47      Assert.assertThat(content, Matchers.containsString("a,3,b,4"));
48      TSDBTxtMeasurementStore store2 = new TSDBTxtMeasurementStore(file);
49      store2.saveMeasurements(id, System.currentTimeMillis(), 5L, 6L);
50      store2.close();
51      content = Files.lines(tmpFile, Charsets.UTF_8).collect(Collectors.joining("\n"));
52      Assert.assertThat(content, Matchers.containsString("a,3,b,4"));
53      Assert.assertThat(content, Matchers.containsString("a,5,b,6"));
54    }
