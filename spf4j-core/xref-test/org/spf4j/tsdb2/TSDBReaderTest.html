72    @Test
73    public void testTailing() throws IOException, InterruptedException, ExecutionException, TimeoutException {
74      File TEST_FILE = File.createTempFile("test", ".tsdb2");
75      try (TSDBWriter writer = new TSDBWriter(TEST_FILE, 4, "test", true);
76              TSDBReader reader = new TSDBReader(TEST_FILE, 1024)) {
77        writer.flush();
78        final BlockingQueue<Either<TableDef, DataBlock>> queue = new ArrayBlockingQueue<>(100);
79        Future<Void> bgWatch = reader.bgWatch((Either<TableDef, DataBlock> object, long deadline) -> {
80          queue.put(object);
81        }, TSDBReader.EventSensitivity.HIGH);
82        long tableId = writer.writeTableDef(tableDef);
83        writer.flush();
84        final long time = System.currentTimeMillis();
85        writer.writeDataRow(tableId, time, 0, 1, 2);
86        writer.writeDataRow(tableId, time + 10, 1, 1, 2);
87        writer.flush();
88        Either<TableDef, DataBlock> td = queue.take();
89        Assert.assertEquals(tableDef, td.getLeft());
90        Either<TableDef, DataBlock> take = queue.take();
91        Assert.assertEquals(2, take.getRight().getValues().size());
92        writer.writeDataRow(tableId, time + 20, 2, 1, 2);
93        writer.writeDataRow(tableId, time + 30, 3, 1, 2);
94        writer.writeDataRow(tableId, time + 40, 4, 1, 2);
95        writer.flush();
96        Assert.assertEquals(3, queue.take().getRight().getValues().size());
97        reader.stopWatching();
98        bgWatch.get(10000, TimeUnit.MILLISECONDS);
99  
100     }
