package org.spf4j.base;

/**
 *
 * @author zoly
 */
public final  class MutableHolder<T> {
    
    private T value;

    public MutableHolder(final T value) {
        this.value = value;
    }

    public MutableHolder() {
        this.value = null;
    }

    public T getValue() {
        return value;
    }

    public void setValue(final T value) {
        this.value = value;
    }

    @Override
    public String toString() {
        return "MutableHolder{" + "value=" + value + '}';
    }
    
}
