16    public static int compareTo(@Nonnull final CharSequence s, @Nonnull final CharSequence t) {
17      int i = 0;
18      final int sl = s.length();
19      final int tl = t.length();
20      while (i < sl && i < tl) {
21        char a = s.charAt(i);
22        char b = t.charAt(i);
23        int diff = a - b;
24        if (diff != 0) {
25          return diff;
26        }
27        i++;
28      }
29      return sl - tl;
30    }

32    public static boolean equalsNullables(@Nullable final CharSequence s, @Nullable final CharSequence t) {
33      if (s == null) {
34        return null == t;
35      } else if (t == null) {
36        return true;
37      } else {
38        return equals(s, t);
39      }
40    }

42    public static boolean equals(@Nonnull final CharSequence s, @Nonnull final CharSequence t) {
43      final int sl = s.length();
44      final int tl = t.length();
45      if (sl != tl) {
46        return false;
47      } else {
48        for (int i = 0; i < sl; i++) {
49          if (s.charAt(i) != t.charAt(i)) {
50            return false;
51          }
52        }
53        return true;
54      }
55    }

57    public static int hashcode(@Nonnull final CharSequence cs) {
58      if (cs instanceof String) {
59        return ((String) cs).hashCode();
60      }
61      int h = 0;
62      int len = cs.length();
63      if (len > 0) {
64        int off = 0;
65        for (int i = 0; i < len; i++) {
66          h = 31 * h + cs.charAt(off++);
67        }
68      }
69      return h;
70    }

72    public static CharSequence subSequence(@Nonnull final CharSequence seq, final int startIdx, final int endIdx) {
73      if (startIdx == 0 && endIdx == seq.length()) {
74        return seq;
75      } else if (startIdx >= endIdx) {
76        return "";
77      } else {
78        return new SubSequence(seq, endIdx - startIdx, startIdx);
79      }
80    }

82    private static final class SubSequence implements CharSequence {
83  
84      private final CharSequence underlyingSequence;
85      private final int length;
86      private final int startIdx;
87  
88      SubSequence(final CharSequence underlyingSequence, final int length, final int startIdx) {
89        this.underlyingSequence = underlyingSequence;
90        this.length = length;
91        this.startIdx = startIdx;
92      }
93  
94      @Override
95      public int length() {
96        return length;
97      }
98  
99      @Override
100     public char charAt(final int index) {
101       return underlyingSequence.charAt(startIdx + index);
102     }
103 
104     @Override
105     public CharSequence subSequence(final int start, final int end) {
106       return CharSequences.subSequence(underlyingSequence, startIdx + start, startIdx + end);
107     }
108 
109     @Override
110     public String toString() {
111       char[] chars = new char[length];
112       int idx = startIdx;
113       for (int i = 0; i < length; i++, idx++) {
114         chars[i] = underlyingSequence.charAt(idx);
115       }
116       return Strings.wrap(chars);
117     }

121   public static boolean endsWith(final CharSequence qc, final CharSequence with) {
122     int l = qc.length();
123     int start = l - with.length();
124     if (start >= 0) {
125       for (int i = start, j = 0; i < l; i++, j++) {
126         if (qc.charAt(i) != with.charAt(j)) {
127           return false;
128         }
129       }
130       return true;
131     } else {
132       return false;
133     }
134   }

141   public static Appendable lineNumbered(final int startLineNr, final Appendable appendable, final IntAppender ia)
142           throws IOException {
143     ia.append(startLineNr, appendable);
144     return new Appendable() {
145       private int lineNr = startLineNr + 1;
146       @Override
147       public Appendable append(final CharSequence csq) throws IOException {
148         return append(csq, 0, csq.length());
149       }
150 
151       @Override
152       public Appendable append(final CharSequence csq, final int start, final int end) throws IOException {
153         int lastIdx = start;
154         for (int i = start; i < end; i++) {
155           if (csq.charAt(i) == '\n') {
156             int next = i + 1;
157             appendable.append(csq, lastIdx, next);
158             ia.append(lineNr++, appendable);
159             lastIdx = next;
160           }
161         }
162         if (lastIdx < end) {
163           appendable.append(csq, lastIdx, end);
164         }
165         return this;
166       }
167 
168       @Override
169       public Appendable append(final char c) throws IOException {
170         appendable.append(c);
171         if (c == '\n') {
172           ia.append(lineNr++, appendable);
173         }
174         return this;
175       }
176     };
177   }

184   public static CharSequence toLineNumbered(final int startLineNr, final CharSequence source,  final IntAppender ia) {
185     int length = source.length();
186     StringBuilder destination = new StringBuilder(length + 6 * length / 80);
187     try {
188       lineNumbered(startLineNr, destination, ia).append(source);
189     } catch (IOException ex) {
190       throw new RuntimeException(ex);
191     }
192     return destination;
