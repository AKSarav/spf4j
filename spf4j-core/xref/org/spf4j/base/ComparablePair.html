19  package org.spf4j.base;
20  
21  import java.io.Serializable;
22  
23  /**
24   *
25   * @author zoly
26   */
27  public final class ComparablePair<A extends Comparable & Serializable, B extends Comparable & Serializable>
28      extends Pair<A, B>
29      implements Comparable<ComparablePair<A, B>>, Serializable  {
30  
31      public ComparablePair(final A first, final B second) {
32          super(first, second);
33      }
34      
35      @edu.umd.cs.findbugs.annotations.SuppressWarnings("UMTP_UNBOUND_METHOD_TEMPLATE_PARAMETER")
36      public static <A extends Comparable & Serializable, B extends Comparable & Serializable>
37                      ComparablePair<A, B> of(final A first, final B second) {
38          return new ComparablePair<A, B>(first, second);
39      }
40  
41      @Override
42      @edu.umd.cs.findbugs.annotations.SuppressWarnings
43      public int compareTo(final ComparablePair<A, B> o) {
44          if (this.first.equals(o.first)) {
45              return this.second.compareTo(o.second);
46          } else {
47              return this.first.compareTo(o.first);
48          }
49      }
50      
51  }
