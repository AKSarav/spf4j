4   import java.lang.ref.SoftReference;
5   
6   /**
7    *
8    * @author zoly
9    */
10  public final class TLScratch {
11  
12    private TLScratch() { }
13  
14    private static final ThreadLocal<SoftReference<byte[]>> BYTES_TMP = new ThreadLocal<>();
15  
16    private static final ThreadLocal<SoftReference<char[]>> CHARS_TMP = new ThreadLocal<>();
17  
18    /**
19     * returns a thread local byte array of at least the size requested. use only for temporary purpose. This method needs
20     * to be carefully used!
21     *
22     * @param size - the minimum size of the temporary buffer requested.
23     * @return - the temporary buffer.
24     */
25    public static byte[] getBytesTmp(final int size) {
26      SoftReference<byte[]> sr = BYTES_TMP.get();
27      byte[] result;
28      if (sr == null) {
29        result = new byte[size];
30        BYTES_TMP.set(new SoftReference<>(result));
31      } else {
32        result = sr.get();
33        if (result == null || result.length < size) {
34          result = new byte[size];
35          BYTES_TMP.set(new SoftReference<>(result));
36        }
37      }
38      return result;
39    }
40  
41    /**
42     * returns a thread local char array of at least the requested size. Use only for temporary purpose.
43     *
44     * @param size - the minimum size of the temporary buffer requested.
45     * @return - the temporary buffer.
46     */
47    public static char[] getCharsTmp(final int size) {
48      SoftReference<char[]> sr = CHARS_TMP.get();
49      char[] result;
50      if (sr == null) {
51        result = new char[size];
52        CHARS_TMP.set(new SoftReference<>(result));
53      } else {
54        result = sr.get();
55        if (result == null || result.length < size) {
56          result = new char[size];
57          CHARS_TMP.set(new SoftReference<>(result));
58        }
59      }
60      return result;
61    }
