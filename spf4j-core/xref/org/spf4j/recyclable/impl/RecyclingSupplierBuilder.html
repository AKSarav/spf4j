18  package org.spf4j.recyclable.impl;
19  
20  import org.spf4j.base.AbstractRunnable;
21  import org.spf4j.recyclable.ObjectCreationException;
22  import org.spf4j.recyclable.ObjectDisposeException;
23  import org.spf4j.recyclable.RecyclingSupplier;
24  import org.spf4j.recyclable.Scanable;
25  import java.util.concurrent.ScheduledExecutorService;
26  import java.util.concurrent.TimeUnit;
27  import org.spf4j.base.Handler;
28  
29  /**
30   *
31   * @author zoly
32   */
33  public final class RecyclingSupplierBuilder<T> {
34  
35      private final int maxSize;
36      private final RecyclingSupplier.Factory<T> factory;
37      private boolean fair;
38      private ScheduledExecutorService maintenanceExecutor;
39      private long maintenanceIntervalMillis;
40      private Handler<T, ? extends Exception> borrowHook;
41      private Handler<T, ? extends Exception> returnHook;
42      private int initialSize;
43      private boolean collectBorrowed;
44  
45      public RecyclingSupplierBuilder(final int maxSize, final RecyclingSupplier.Factory<T> factory) {
46          this.fair = true;
47          this.maxSize = maxSize;
48          this.factory = factory;
49          this.initialSize = 0;
50      }
51  
52      public RecyclingSupplierBuilder<T> unfair() {
53          this.fair = false;
54          return this;
55      }
56  
57      public RecyclingSupplierBuilder<T> withInitialSize(final int pinitialSize) {
58          this.initialSize = pinitialSize;
59          return this;
60      }
61      
62  
63      public RecyclingSupplierBuilder<T> withMaintenance(final ScheduledExecutorService pexec,
64              final long pmaintenanceIntervalMillis, final boolean pcollectBorrowed) {
65          this.maintenanceExecutor = pexec;
66          this.maintenanceIntervalMillis = pmaintenanceIntervalMillis;
67          this.collectBorrowed = pcollectBorrowed;
68          return this;
69      }
70  
71      public RecyclingSupplierBuilder<T> withGetHook(final Handler<T, ? extends Exception> phook) {
72          this.borrowHook = phook;
73          return this;
74      }
75  
76      public RecyclingSupplierBuilder<T> withRecycleHook(final Handler<T, ? extends Exception> phook) {
77          this.returnHook = phook;
78          return this;
79      }
80  
81      public RecyclingSupplier<T> build() throws ObjectCreationException {
82          final ScalableObjectPool<T> underlyingPool =
83                  new ScalableObjectPool<>(initialSize, maxSize, factory, fair);
84          final RecyclingSupplier<T> pool;
85          if (borrowHook != null || returnHook != null) {
86              pool = new ObjectPoolWrapper<>(underlyingPool, borrowHook, returnHook);
87          } else {
88              pool = underlyingPool;
89          }
90          if (maintenanceExecutor != null) {
91              maintenanceExecutor.scheduleWithFixedDelay(new AbstractRunnableImpl<>(underlyingPool, collectBorrowed),
92                      maintenanceIntervalMillis, maintenanceIntervalMillis, TimeUnit.MILLISECONDS);
93          }
94          return pool;
95      }
96  
97      public static final class AbstractRunnableImpl<T> extends AbstractRunnable {
98  
99          private final ScalableObjectPool<T> underlyingPool;
100         private final boolean collectBorrowed;
101 
102         public AbstractRunnableImpl(final ScalableObjectPool<T> underlyingPool, final boolean collectBorrowed) {
103             super(true);
104             this.underlyingPool = underlyingPool;
105             this.collectBorrowed = collectBorrowed;
106         }
107 
108         @Override
109         public void doRun() throws Exception {
110                 if (collectBorrowed) {
111                     underlyingPool.requestReturnFromBorrowersIfNotInUse();
112                 }
113                 underlyingPool.scan(new Scanable.ScanHandler<ObjectHolder<T>>() {
114                     @Override
115                     public boolean handle(final ObjectHolder<T> object) throws ObjectDisposeException {
116                         object.validateObjectIfNotBorrowed();
117                         return true;
118                     }
119                 });
120            
121         }
122     }
123 }
