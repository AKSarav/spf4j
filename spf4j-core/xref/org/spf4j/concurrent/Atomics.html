3   import java.util.Objects;
4   import java.util.concurrent.atomic.AtomicReference;
5   import java.util.function.UnaryOperator;
6   import javax.annotation.ParametersAreNonnullByDefault;
7   
8   /**
9    *
10   * @author zoly
11   */
12  @ParametersAreNonnullByDefault
13  public final class Atomics {
14  
15    private Atomics() {
16    }
17  
18    public static <T> UpdateResult<T> update(final AtomicReference<T> ar, final UnaryOperator<T> function) {
19      T initial;
20      T newObj;
21      do {
22        initial = ar.get();
23        newObj = function.apply(initial);
24        if (Objects.equals(initial, newObj)) {
25          return UpdateResult.same(initial);
26        } else if (initial == newObj) {
27          throw new IllegalStateException("Function " + function + " is mutating " + initial + ", this is not allowed");
28        }
29      } while (!ar.compareAndSet(initial, newObj));
30      return UpdateResult.updated(newObj);
31    }
32  
33  }
