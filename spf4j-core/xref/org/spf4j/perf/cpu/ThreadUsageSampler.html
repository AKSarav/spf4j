29    private static final ThreadMXBean TH_BEAN = ManagementFactory.getThreadMXBean();

31    private static ScheduledFuture<?> samplingFuture;

33    private static final List<String> PEAK_THREAD_NAMES = new ArrayList<>();
34    private static final List<StackTraceElement[]> PEAK_THREAD_TRACES = new ArrayList<>();
35    private static final BitSet PEAK_THREAD_DAEMON = new BitSet();

37    static {
38      org.spf4j.base.Runtime.queueHook(2, new AbstractRunnable(true) {
39        @Override
40        public void doRun() {
41          stop();
42          writePeakThreadInfo(System.err);
43        }
44  
45      });
46      Registry.export(ThreadUsageSampler.class);
47    }

49    private ThreadUsageSampler() {

52    public static void writePeakThreadInfo(final PrintStream out) {
53      out.println("Peak Threads:");
54      int i = 0;
55      boolean haveStacktraces = PEAK_THREAD_TRACES.size() > 0;
56      for (String tname : PEAK_THREAD_NAMES) {
57        out.print(tname);
58        out.print(", daemon =");
59        out.print(PEAK_THREAD_DAEMON.get(i));
60        out.print(',');
61        if (haveStacktraces) {
62          out.print(' ');
63          out.print(Arrays.toString(PEAK_THREAD_TRACES.get(i)));
64        }
65        out.println();
66        i++;
67      }
68    }

70    @JmxExport
71    @SuppressFBWarnings({"DM_DEFAULT_ENCODING", "NP_LOAD_OF_KNOWN_NULL_VALUE"})
72    public static String getPeakThreadInfo() {
73      try (ByteArrayBuilder bab = new ByteArrayBuilder()) {
74        PrintStream ps = new PrintStream(bab);
75        writePeakThreadInfo(ps);
76        return bab.toString(Charset.defaultCharset());
77      }
78    }

80    @JmxExport
81    public static String getCurrentAliveThreadInfo() {
82      StringBuilder sb = new StringBuilder(2048);
83      Thread[] threads = FastStackCollector.getThreads();
84      StackTraceElement[][] stackTraces = FastStackCollector.getStackTraces(threads);
85      for (int i = 0; i < threads.length; i++) {
86        Thread t = threads[i];
87        if (t.isAlive()) {
88          sb.append(t.getId());
89          sb.append(",\t").append(t.getName());
90          sb.append(",\t state =").append(t.getState());
91          sb.append(",\t daemon =").append(t.isDaemon());
92          sb.append(",\t");
93          StackTraceElement[] straces = stackTraces[i];
94          if (straces != null && straces.length > 0) {
95            sb.append(' ');
96            sb.append(Arrays.toString(straces));
97          }
98          sb.append('\n');
99        }
100     }
101     return sb.toString();
102   }

105   public static synchronized void start(final int sampleTime) {
106     start(sampleTime, true);
107   }

109   @JmxExport
110   public static synchronized void start(@JmxExport("sampleTimeMillis") final int sampleTime,
111           @JmxExport("withStackTraces") final boolean withStackTraces) {
112     if (samplingFuture == null) {
113       final MeasurementRecorder cpuUsage
114               = RecorderFactory.createDirectRecorder("peak-thread-count", "count", sampleTime);
115       samplingFuture = DefaultScheduler.INSTANCE.scheduleWithFixedDelay(new AbstractRunnable() {
116 
117         private int maxThreadsNr = 0;
118 
119         @Override
120         public void doRun() {
121           final int peakThreadCount = TH_BEAN.getPeakThreadCount();
122           cpuUsage.record(peakThreadCount);
123           if (peakThreadCount > maxThreadsNr) {
124             Thread[] ths = FastStackCollector.getThreads();
125             if (ths.length > PEAK_THREAD_NAMES.size()) {
126               if (withStackTraces) {
127                 StackTraceElement[][] stackTraces = FastStackCollector.getStackTraces(ths);
128                 PEAK_THREAD_TRACES.clear();
129                 PEAK_THREAD_TRACES.addAll(Arrays.asList(stackTraces));
130               }
131 
132               PEAK_THREAD_NAMES.clear();
133               int i = 0;
134               for (Thread th : ths) {
135                 PEAK_THREAD_NAMES.add(th.getName());
136                 if (th.isDaemon()) {
137                   PEAK_THREAD_DAEMON.set(i);
138                 } else {
139                   PEAK_THREAD_DAEMON.clear(i);
140                 }
141                 i++;
142               }
143             }
144             maxThreadsNr = peakThreadCount;
145           }
146           TH_BEAN.resetPeakThreadCount();
147         }
148       }, sampleTime, sampleTime, TimeUnit.MILLISECONDS);
149     } else {
150       throw new IllegalStateException("Thread sampling already started " + samplingFuture);
151     }
152   }
153 
154   @JmxExport
155   public static synchronized void stop() {
156     if (samplingFuture != null) {
157       samplingFuture.cancel(false);
158       samplingFuture = null;
159     }
160   }
161 
162   @JmxExport
163   public static synchronized boolean isStarted() {
164     return samplingFuture != null;
165   }
