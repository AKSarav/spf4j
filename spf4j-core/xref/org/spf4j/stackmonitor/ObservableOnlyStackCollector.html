18  package org.spf4j.stackmonitor;
19  
20  import com.google.common.annotations.Beta;
21  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
22  import java.util.concurrent.ConcurrentHashMap;
23  import java.util.concurrent.ConcurrentMap;
24  import java.util.function.Consumer;
25  import javax.annotation.Nullable;
26  import org.spf4j.base.ArrayBuilder;
27  
28  @Beta
29  public final class ObservableOnlyStackCollector extends AbstractStackCollector {
30  
31    private final ConcurrentMap<Thread, Consumer<StackTraceElement[]>> stConsumers;
32  
33  
34    public ObservableOnlyStackCollector() {
35      this.stConsumers = new ConcurrentHashMap<>();
36    }
37  
38    @Nullable
39    public Consumer<StackTraceElement[]> registerConsumer(final Thread thread,
40            final Consumer<StackTraceElement[]> consumer) {
41      return stConsumers.put(thread, consumer);
42    }
43  
44    public Consumer<StackTraceElement[]> unregisterConsumer(final Thread thread) {
45      return stConsumers.remove(thread);
46    }
47  
48    private ArrayBuilder<Thread> requestFor = new ArrayBuilder<>(8, Thread.class);
49  
50    @Override
51    @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CHECKED")
52    public void sample(final Thread ignore) {
53      requestFor.clear();
54      for (Thread t : stConsumers.keySet()) {
55        requestFor.add(t);
56      }
57      Thread[] rfArray = requestFor.getArray();
58      StackTraceElement[][] stackDump = FastStackCollector.getStackTraces(rfArray);
59      int size = requestFor.getSize();
60      for (int i = 0; i < size; i++) {
61        Thread t = rfArray[i];
62        StackTraceElement[] stackTrace = stackDump[i];
63        Consumer<StackTraceElement[]> consumer = this.stConsumers.get(t);
64        if (consumer != null) {
65          consumer.accept(stackTrace);
66        }
67        if (stackTrace != null && stackTrace.length > 0) {
68          addSample(stackTrace);
69        } else {
70          addSample(new StackTraceElement[]{
71            new StackTraceElement("Thread", t.getName(), "", 0)
72          });
73        }
74      }
75    }
76  
77  }
