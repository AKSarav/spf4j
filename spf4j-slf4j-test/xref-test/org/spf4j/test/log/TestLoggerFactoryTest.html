54        logMarkerTests();

62    public static void logTests() {
63      LOG.trace("Hello logger", new RuntimeException());
64      LOG.trace("Hello logger");
65      LOG.trace("Hello logger {}", 1);
66      LOG.trace("Hello logger {} {} {}", 1, 2, 3);
67      LOG.trace("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
68      LOG.debug("Hello logger", new RuntimeException());
69      LOG.debug("Hello logger");
70      LOG.debug("Hello logger {}", 1);
71      LOG.debug("Hello logger {} {} {}", 1, 2, 3);
72      LOG.debug("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
73      LOG.info("Hello logger", new RuntimeException());
74      LOG.info("Hello logger");
75      LOG.info("Hello logger {}", 1);
76      LOG.info("Hello logger {} {} {}", 1, 2, 3);
77      LOG.info("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
78      LOG.warn("Hello logger", new RuntimeException());
79      LOG.warn("Hello logger");
80      LOG.warn("Hello logger {}", 1);
81      LOG.warn("Hello logger {} {} {}", 1, 2, 3);
82      LOG.warn("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
83      LogAssert expect = TestLoggers.sys().expect("org.spf4j.test", Level.ERROR, 5,
84              LogMatchers.hasMatchingFormat(Matchers.containsString("Hello logger")));
85      LOG.error("Hello logger", new RuntimeException());
86      LOG.error("Hello logger");
87      LOG.error("Hello logger {}", 1);
88      LOG.error("Hello logger {} {} {}", 1, 2, 3);
89      LOG.error("Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
90      expect.assertObservation();
91    }

93    public static void logMarkerTests() {
94      Marker marker = MarkerFactory.getMarker("TEST");
95      Marker marker2 = MarkerFactory.getMarker("TEST2");
96      marker2.add(marker);
97      marker2.add(MarkerFactory.getMarker("TEST3"));
98  
99      LOG.trace(marker, "Hello logger", new RuntimeException());
100     LOG.trace(marker, "Hello logger");
101     LOG.trace(marker, "Hello logger {}", 1);
102     LOG.trace(marker, "Hello logger {} {} {}", 1, 2, 3);
103     LOG.trace(marker, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
104     LOG.debug(marker, "Hello logger", new RuntimeException());
105     LOG.debug(marker, "Hello logger");
106     LOG.debug(marker, "Hello logger {}", 1);
107     LOG.debug(marker, "Hello logger {} {} {}", 1, 2, 3);
108     LOG.debug(marker, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
109     LOG.info(marker, "Hello logger", new RuntimeException());
110     LOG.info(marker2, "Hello logger");
111     LOG.info(marker2, "Hello logger {}", 1);
112     LOG.info(marker2, "Hello logger {} {} {}", 1, 2, 3);
113     LOG.info(marker2, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
114     LOG.warn(marker, "Hello logger", new RuntimeException());
115     LOG.warn(marker, "Hello logger");
116     LOG.warn(marker, "Hello logger {}", 1);
117     LOG.warn(marker, "Hello logger {} {} {}", 1, 2, 3);
118     LOG.warn(marker, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
119     LogAssert expect = TestLoggers.sys().expect("org.spf4j.test", Level.ERROR, 5,
120             Matchers.allOf(
121                     LogMatchers.hasMatchingFormat(Matchers.containsString("Hello logger")),
122                     LogMatchers.hasMarker(marker)));
123     LOG.error(marker, "Hello logger", new RuntimeException());
124     LOG.error(marker, "Hello logger");
125     LOG.error(marker, "Hello logger {}", 1);
126     LOG.error(marker, "Hello logger {} {} {}", 1, 2, 3);
127     LOG.error(marker, "Hello logger {} {} {}", 1, 2, 3, new RuntimeException());
128     expect.assertObservation();
129   }

132   @SuppressFBWarnings("LO_TOSTRING_PARAMETER") // this is on purpose configs change over time.
133   public void testIsEnabled() {
134     LOG.debug("Log Config: {}", TestLoggers.sys().toString());
135     Marker marker = MarkerFactory.getMarker("TEST");
136     Assert.assertFalse(LOG.isTraceEnabled());
137     Assert.assertFalse(LOG.isTraceEnabled(marker));
138 
139     //DEBUG is enabled since debug logs are collected in the background...
140     Assert.assertTrue(LOG.isDebugEnabled());
141     Assert.assertTrue(LOG.isDebugEnabled(marker));
142 
143     Assert.assertTrue(LOG.isInfoEnabled());
144     Assert.assertTrue(LOG.isInfoEnabled(marker));
145     Assert.assertTrue(LOG.isWarnEnabled());
146     Assert.assertTrue(LOG.isWarnEnabled(marker));
147     Assert.assertTrue(LOG.isErrorEnabled());
148     Assert.assertTrue(LOG.isErrorEnabled(marker));
149   }

239   @Test
240   @SuppressFBWarnings("UTAO_JUNIT_ASSERTION_ODDITIES_NO_ASSERT")
241   public void testUncaught() throws InterruptedException {
242     IllegalStateException ex = new IllegalStateException();
243     Thread thread = new Thread(() -> {
244       throw ex;
245     });
246     TestLoggers config = TestLoggers.sys();
247     AsyncObservationAssert obs = config
248             .expectUncaughtException(
249                     UncaughtExceptionDetail.hasThrowable(Matchers.equalTo(ex)));
250     thread.start();
251     obs.assertObservation(5, TimeUnit.SECONDS);
252   }
