93      public BigInteger fibB(final int i) {
94          if (i <= 1) {
95              return BigInteger.valueOf(i);
96          } else {
97              return fibB(i - 1).add(fibB(i - 2));
98          }
99      }
100 
101     public long fib(final long i) {
102         if (i <= 1) {
103             return i;
104         } else {
105             return fib(i - 1) + fib(i - 2);
106         }
107     }
108 
109     public BigInteger fibBNr(final int i) {
110         if (i <= 1) {
111             return BigInteger.valueOf(i);
112         } else {
113             BigInteger v1 = BigInteger.ZERO;
114             BigInteger v2 = BigInteger.ONE;
115             BigInteger tmp;
116             for (int j = 2; j <= i; j++) {
117                 tmp = v2;
118                 v2 = v1.add(v2);
119                 v1 = tmp;
120             }
121             return v2;
122         }
123     }

125     @Test
126     public void testFibPerformance() throws CompileException, ExecutionException, InterruptedException {
127         String fib = "func det fib (x) {fib(x-1) + fib(x-2)}; fib(0) = 0; fib(1) = 1; fib(x)";
128         Program fibZel = Program.compile(fib, "x");
129         long startTime = System.currentTimeMillis();
130         Number zelResult = (Number) fibZel.execute(40);
131         long intTime = System.currentTimeMillis();
132         long javaResult = fib(40);
133         long stopTime = System.currentTimeMillis();
134         Assert.assertEquals(zelResult.longValue(), javaResult);
135         System.out.println("zel exec " + (intTime - startTime));
136         System.out.println("java exec " + (stopTime - intTime));
137 
138         startTime = System.currentTimeMillis();
139         BigInteger zelResult2 = (BigInteger) fibZel.execute(10000);
140         intTime = System.currentTimeMillis();
141         BigInteger javaResult2 = fibBNr(10000);
142         stopTime = System.currentTimeMillis();
143         System.out.println("fib(10000) = " + zelResult2);
144         System.out.println("fib(10000) = " + javaResult2);
145         Assert.assertEquals(zelResult2, javaResult2);
146         System.out.println("zel exec " + (intTime - startTime));
147         System.out.println("java exec " + (stopTime - intTime));
148     }

181     @Test
182     public void testCond() throws CompileException, ExecutionException, InterruptedException {
183         Boolean result = (Boolean) Program.compile("x == 1", "x").execute(1);
184         Assert.assertTrue(result);
185         result = (Boolean) Program.compile("x != 1", "x").execute(1);
186         Assert.assertFalse(result);
187         result = (Boolean) Program.compile("x < 1", "x").execute(1);
188         Assert.assertFalse(result);
189         result = (Boolean) Program.compile("x >= 1", "x").execute(1);
190         Assert.assertTrue(result);
191         result = (Boolean) Program.compile("x <= 1", "x").execute(1);
192         Assert.assertTrue(result);
193         result = (Boolean) Program.compile("x > 1", "x").execute(1);
194         Assert.assertFalse(result);
195         Number n = (Number) Program.compile("min(3, 1, 8)").execute();
196         Assert.assertEquals(1, n.intValue());
197         n = (Number) Program.compile("max(3, 1, 8, 10)").execute();
198         Assert.assertEquals(10, n.intValue());
199         n = (Number) Program.compile("sqrt(4)").execute();
200         Assert.assertEquals(2, n.intValue());
201         n = (Number) Program.compile("log(E)").execute();
202         Assert.assertEquals(1, n.intValue());
203

208         String prog = "f = func (a, b) {sleep 1000; a + b };"

216         Assert.assertEquals(10, result.intValue());
