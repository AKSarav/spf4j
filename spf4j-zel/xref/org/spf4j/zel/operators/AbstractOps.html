4   import java.util.HashMap;
5   import java.util.Map;
6   
7   /**
8    *
9    * @author zoly
10   */
11  abstract class AbstractOps<LT> implements Operator<LT, Number, Number> {
12  
13          //CHECKSTYLE:OFF
14          final Map<Class, Operator<LT, Number, Number>> operations;
15          //CHECKSTYLE:ON
16  
17          public AbstractOps() {
18              operations = new HashMap<>();
19          }
20  
21          @Override
22          public  Number op(final LT a, final Number b) {
23              final Operator<LT, Number, Number> op = operations.get(b.getClass());
24              if (op == null) {
25                  throw new UnsupportedOperationException("Not supporting " + this + " for " + a + " and " + b);
26              }
27              return op.op(a, b);
28          }
29  
30      }
