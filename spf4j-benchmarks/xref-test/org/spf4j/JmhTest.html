21      @Test
22      public void runJmh() throws RunnerException, IOException {
23          final String destinationFolder = System.getProperty("basedir",
24                  org.spf4j.base.Runtime.USER_DIR) + "/target";
25          final String profile = System.getProperty("basedir",
26                  org.spf4j.base.Runtime.USER_DIR) + "/src/main/jfc/profile.jfc";
27          Options opt = new OptionsBuilder()
28                  //.include(".*Appendable.*")
29                  .include(".*")
30                  .addProfiler(JmhProfiler.class)
31  //                .addProfiler(CompilerProfiler.class)
32                  .addProfiler(JmhFlightRecorderProfiler.class)
33  //                .addProfiler(GCProfiler.class)
34                  //"-XX:+PrintCompilation", "-XX:+UseG1GC",
35                  .jvmArgs("-XX:+UseG1GC", "-Xmx256m", "-Xms256m", "-XX:+UnlockCommercialFeatures",
36                          "-Djmh.stack.profiles=" + destinationFolder,
37                          "-Dspf4j.executors.defaultExecutor.daemon=true", "-Djmh.executor=FJP",
38                          "-Djmh.fr.options=defaultrecording=true,settings=" + profile)
39                  .result(destinationFolder + "/" + "benchmarkResults.csv")
40                  .resultFormat(ResultFormatType.CSV)
41                  .warmupIterations(10)
42                  .measurementIterations(10)
43                  .forks(1)
44                  .build();
45           new Runner(opt).run();
46      }
