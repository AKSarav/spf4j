16  @Fork(2)
17  @Threads(value = 8)
18  public class AppendableWriterBenchmark {
19  
20    @State(Scope.Benchmark)
21    public static class BenchmarkState {
22  
23      public final char[] testChars;
24  
25      {
26        StringBuilder builder = new StringBuilder(4096);
27        for (int i = 0; i < 4096; i++) { // simulating a 8k buffer.
28          builder.append('A');
29        }
30        testChars = builder.toString().toCharArray();
31      }
32  
33    }
34  
35    @State(Scope.Thread)
36    public static class ThreadState {
37      public final StringBuilder sb = new StringBuilder(100000);
38      public final IntMath.XorShift32 random = new IntMath.XorShift32();
39    }
40  
41    @Benchmark
42    public final StringBuilder guavaAppendable(final BenchmarkState bs, final ThreadState ts) throws IOException {
43      StringBuilder stringBuilder = ts.sb;
44      stringBuilder.setLength(0);
45      Writer writer = new AppendableWriter(stringBuilder);
46      for (int i = 0; i < 25; i++) {
47        writer.write(bs.testChars, 0, Math.abs(ts.random.nextInt()) % 4096);
48      }
49      writer.close();
50      return stringBuilder;
51    }
52  
53    @Benchmark
54    public final StringBuilder spf4jAppendable(final BenchmarkState bs, final ThreadState ts) throws IOException {
55      StringBuilder stringBuilder = ts.sb;
56      stringBuilder.setLength(0);
57      Writer writer = new org.spf4j.io.AppendableWriter(stringBuilder);
58      for (int i = 0; i < 25; i++) {
59        writer.write(bs.testChars, 0, Math.abs(ts.random.nextInt()) % 4096);
60      }
61      writer.close();
62      return stringBuilder;
63    }
