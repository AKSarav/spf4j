2   package org.spf4j.base.asm;
3   
4   import java.lang.reflect.Method;
5   import java.util.ArrayDeque;
6   import java.util.Arrays;
7   import java.util.Collection;
8   import java.util.Deque;
9   import java.util.HashMap;
10  import java.util.Map;
11  import java.util.Set;
12  import org.objectweb.asm.Label;
13  import org.objectweb.asm.MethodVisitor;
14  import org.objectweb.asm.Type;
15  import org.objectweb.asm.commons.EmptyVisitor;
16  
17  /**
18   * The purpose of this visitor is to find all incovation of certain method.
19   * Also to try to get the invocation parameters in case they are know at compile time.
20   * Since not all stack operations are simulated yet, the invocation parameters might not be accurate.
21   *
22   * @author zoly
23   */
24  class MethodInvocationClassVisitor extends EmptyVisitor {
25  
26      private final Collection<Invocation> addCaleesTo;
27      private final Map<String, Method> methodStrings;
28      private String className;
29      private String source;
30  
31      MethodInvocationClassVisitor(final Collection addCaleesTo,
32              final Set<Method> methods) {
33          this.addCaleesTo = addCaleesTo;
34          this.methodStrings = new HashMap<>(methods.size());
35          for (Method m : methods) {
36              this.methodStrings.put(TypeUtils.toString(m), m);
37          }
38      }
39  
40      @Override
41      public void visit(final int version, final int access, final String name,
42              final String signature, final String superName, final String[] interfaces) {
43          className = name;
44      }
45  
46      @Override
47      public void visitSource(final String psource, final String debug) {
48          this.source = psource;
49      }
50  
51      @Override
52      public MethodVisitor visitMethod(final int access, final String methodName,
53              final String methodDesc, final String signature,
54              final String[] exceptions) {
55          return new EmptyVisitor() {
56  
57              private int lineNumber;
58  
59              private final Deque<Object> stack = new ArrayDeque<>();
60  
61              {
62                  Type[] argumentTypes = Type.getArgumentTypes(methodDesc);
63                  stack.addAll(Arrays.asList(argumentTypes));
64              }
65  
66              @Override
67              public void visitMethodInsn(final int opcode, final String owner, final String name, final String desc) {
68                  final Method invokedMethod = methodStrings.get(owner + '/' + name + desc);
69                  boolean isCallingtarget = invokedMethod != null;
70                  Type[] parameterTypes = Type.getArgumentTypes(desc);
71                  Object[] parameters = new Object[parameterTypes.length];
72                  for (int i = parameterTypes.length - 1; i >= 0; i--) {
73                      if (stack.isEmpty()) {
74                          parameters[i] = "UNKNOWN";
75                      } else {
76                          parameters[i] = stack.pop();
77                      }
78                  }
79                  if (isCallingtarget) {
80                      addCaleesTo.add(new Invocation(className, methodName, methodDesc, parameters,
81                              source, lineNumber, invokedMethod));
82                  }
83              }
84  
85              @Override
86              public void visitLdcInsn(final Object o) {
87                  stack.push(o);
88              }
89  
90              @Override
91              public void visitLineNumber(final int line, final Label start) {
92                  this.lineNumber = line;
93              }
94  
95          };
96      }
97  
98      @Override
99      public String toString() {
100         return "MethodInvocationClassVisitor{" + "addCaleesTo=" + addCaleesTo + ", methodStrings=" + methodStrings
101                 + ", className=" + className + ", source=" + source + '}';
102     }
