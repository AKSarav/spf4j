3   import java.lang.reflect.Method;
4   import java.util.Arrays;
5   
6   /**
7    *
8    * @author zoly
9    */
10  public final class Invocation {
11  
12      private final String caleeClassName;
13      private final String caleeMethodName;
14      private final String caleeMethodDesc;
15      private final String caleeSource;
16      private final int caleeLine;
17      private final Object [] parameters;
18      private final Method invokedMethod;
19  
20      public Invocation(final String cName, final String mName, final String mDesc,
21              final Object [] parameters, final String src, final int ln, final Method invokedMethod) {
22          caleeClassName = cName;
23          caleeMethodName = mName;
24          caleeMethodDesc = mDesc;
25          this.parameters = parameters.clone();
26          caleeSource = src;
27          caleeLine = ln;
28          this.invokedMethod = invokedMethod;
29      }
30  
31      public String getCaleeClassName() {
32          return caleeClassName;
33      }
34  
35      public String getCaleeMethodName() {
36          return caleeMethodName;
37      }
38  
39      public String getCaleeMethodDesc() {
40          return caleeMethodDesc;
41      }
42  
43      public String getCaleeSource() {
44          return caleeSource;
45      }
46  
47      public int getCaleeLine() {
48          return caleeLine;
49      }
50  
51      public Method getInvokedMethod() {
52          return invokedMethod;
53      }
54  
55  
56  
57      public Object[] getParameters() {
58          return parameters.clone();
59      }
60  
61      @Override
62      public String toString() {
63          return "Invocation{" + "caleeClassName=" + caleeClassName + ", caleeMethodName="
64                  + caleeMethodName + ", caleeMethodDesc=" + caleeMethodDesc + ", caleeSource="
65                  + caleeSource + ", caleeLine=" + caleeLine + ", parameters=" + Arrays.toString(parameters)
66                  + ", invokedMethod=" + invokedMethod + '}';
67      }

71  
72  }
