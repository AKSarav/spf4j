46      private static final Map<ObjectName, Object> REGISTERED = new HashMap<>();

48      public static synchronized Object registerMBean(final ObjectName objectName, final Object mbean) {
49          Object replaced = null;
50          if (MBEAN_SERVER.isRegistered(objectName)) {
51              try {
52                  replaced = REGISTERED.remove(objectName);
53                  MBEAN_SERVER.unregisterMBean(objectName);
54              } catch (InstanceNotFoundException | MBeanRegistrationException ex) {
55                  throw new RuntimeException(ex);
56              }
57          }
58          try {
59              MBEAN_SERVER.registerMBean(mbean, objectName);
60          } catch (InstanceAlreadyExistsException | MBeanRegistrationException | NotCompliantMBeanException ex) {
61              throw new RuntimeException(ex);
62          }
63          REGISTERED.put(objectName, mbean);
64          return replaced;
65      }

68          return REGISTERED.get(objectName);
69      }

93      public static synchronized Object unregister(final ObjectName objectName) {
94          Object result = null;
95          if (MBEAN_SERVER.isRegistered(objectName)) {
96              try {
97                  result = REGISTERED.remove(objectName);
98                  MBEAN_SERVER.unregisterMBean(objectName);
99              } catch (InstanceNotFoundException | MBeanRegistrationException ex) {
100                 throw new RuntimeException(ex);
101             }
102         }
103         return result;
104     }

126     public static synchronized ExportedValuesMBean export(final String packageName, final String mbeanName,
127             final Map<String, Object> attributes, final Object ... objects) {
128 
129         ObjectName objectName = ExportedValuesMBean.createObjectName(packageName, mbeanName);
130         ExportedValuesMBean existing = (ExportedValuesMBean) unregister(objectName);
131         Map<String, ExportedValue> exportedAttributes = new HashMap<>();
132         if (attributes != null) {
133             for (String attrName : attributes.keySet()) {
134                 exportedAttributes.put(attrName, new MapExportedValue(attributes, null, attrName));
135             }
136         }
137         Map<String, ExportedOperationImpl> exportedOps = new HashMap<>();
138         for (Object object : objects) {
139 
140             if (object instanceof Class) {
141                 for (Method method : ((Class<?>) object).getMethods()) {
142                     if (method.isSynthetic()) {
143                         continue;
144                     }
145                     if (Modifier.isStatic(method.getModifiers())) {
146                         Annotation[] annotations = method.getAnnotations();
147                         for (Annotation annot : annotations) {
148                             if (annot.annotationType() == JmxExport.class) {
149                                 exportMethod(method, null, exportedAttributes, exportedOps, annot);
150                             }
151                         }
152                     }
153                 }
154             } else {
155                 final Class<? extends Object> oClass = object.getClass();
156                 for (Method method : oClass.getMethods()) {
157                     if (method.isSynthetic()) {
158                         continue;
159                     }
160                     Annotation[] annotations = method.getAnnotations();
161                     for (Annotation annot : annotations) {
162                         if (annot.annotationType() == JmxExport.class) {
163                             exportMethod(method, object, exportedAttributes, exportedOps, annot);
164                         }
165                     }
166                 }
167             }
168 
169         }
170         if (exportedAttributes.isEmpty() && exportedOps.isEmpty()) {
171             return null;
172         }
173         ExportedValue<?>[] values = new ExportedValue[exportedAttributes.size()];
174         int i = 0;
175         for (ExportedValue expVal : exportedAttributes.values()) {
176             if (expVal instanceof ExportedValueImpl && !((ExportedValueImpl) expVal).isValid()) {
177                throw new IllegalArgumentException("If setter is exported, getter must be exported as well " + expVal);
178             } else {
179                values[i++] = expVal;
180             }
181         }
182         ExportedValuesMBean mbean;
183         if (existing == null) {
184             mbean = new ExportedValuesMBean(objectName, values,
185                         exportedOps.values().toArray(new ExportedOperation[exportedOps.size()]));
186         } else {
187             mbean = new ExportedValuesMBean(existing, values,
188                         exportedOps.values().toArray(new ExportedOperation[exportedOps.size()]));
189         }
190         Registry.registerMBean(mbean.getObjectName(), mbean);
191         return mbean;
192     }

194     private static void exportMethod(final Method method,
195             @Nullable final Object object, final Map<String, ExportedValue> exportedAttributes,
196             final Map<String, ExportedOperationImpl> exportedOps, final Annotation annot) {
197         method.setAccessible(true); // this is to speed up invocation
198         String methodName = method.getName();
199         int nrParams = method.getParameterTypes().length;
200         if (nrParams == 0 && methodName.startsWith("get")) {
201             String valueName = methodName.substring("get".length());
202             valueName = Strings.withFirstCharLower(valueName);
203             addGetter(valueName, exportedAttributes, annot, method, object);
204         } else if (nrParams == 0 && methodName.startsWith("is")) {
205             String valueName = methodName.substring("is".length());
206             valueName = Strings.withFirstCharLower(valueName);
207             addGetter(valueName, exportedAttributes, annot, method, object);
208         } else if (nrParams == 1 && methodName.startsWith("set")) {
209             addSetter(methodName, exportedAttributes, method, object, annot);
210         } else {
211             String opName = methodName;
212             String nameOverwrite = (String) Reflections.getAnnotationAttribute(annot, "value");
213             if (!"".equals(nameOverwrite)) {
214                 opName = nameOverwrite;
215             }
216             ExportedOperationImpl existing = exportedOps.put(opName, new ExportedOperationImpl(opName,
217                     (String) Reflections.getAnnotationAttribute(annot, "description"), method, object));
218             if (existing != null) {
219                 throw new IllegalArgumentException("exporting operations with same name not supported: " + opName);
220             }
221         }
222     }

224     private static void addSetter(final String methodName, final Map<String, ExportedValue> exportedAttributes,
225             final Method method, final Object object, final Annotation annot) {
226         String customName = (String) Reflections.getAnnotationAttribute(annot, "value");
227         String valueName;
228         if ("".equals(customName)) {
229             valueName = methodName.substring("set".length());
230             valueName = Strings.withFirstCharLower(valueName);
231         } else {
232             valueName = customName;
233         }
234 
235         ExportedValueImpl existing = (ExportedValueImpl) exportedAttributes.get(valueName);
236         if (existing == null) {
237             existing = new ExportedValueImpl(valueName, null,
238                     null, method, object, method.getParameterTypes()[0]);
239         } else {
240             if (existing.getValueClass() != method.getParameterTypes()[0]) {
241                 throw new IllegalArgumentException(
242                         "Getter and setter icorrectly defined " + existing + ' ' + method);
243             }
244             existing = existing.withSetter(method);
245         }
246         exportedAttributes.put(valueName, existing);
247     }
248 
249     private static void addGetter(final String pvalueName,
250             final Map<String, ExportedValue> exported,
251             final Annotation annot, final Method method, final Object object) {
252         String customName = (String) Reflections.getAnnotationAttribute(annot, "value");
253         String valueName = "".equals(customName) ? pvalueName : customName;
254         ExportedValueImpl existing = (ExportedValueImpl) exported.get(valueName);
255         if (existing == null) {
256             existing = new ExportedValueImpl(
257                     valueName, (String) Reflections.getAnnotationAttribute(annot, "description"),
258                     method, null, object, method.getReturnType());
259         } else {
260             if (existing.getValueClass() != method.getReturnType()) {
261                 throw new IllegalArgumentException(
262                         "Getter and setter icorrectly defined " + existing + ' ' + method);
263             }
264             existing = existing.withGetter(method,
265                     (String) Reflections.getAnnotationAttribute(annot, "description"));
266         }
267         exported.put(valueName, existing);
268     }
