3   import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
4   import java.io.File;
5   import java.io.IOException;
6   import java.util.concurrent.ExecutionException;
7   import java.util.concurrent.TimeoutException;
8   import org.junit.runner.Description;
9   import org.junit.runner.notification.Failure;
10  import org.junit.runner.notification.RunListener;
11  import org.spf4j.stackmonitor.FastStackCollector;
12  import org.spf4j.stackmonitor.Sampler;
13  
14  /**
15   * A Junit run listener that will collect performance profiles for your unit tests.
16   *
17   * Works only if unit tests are NOT executed in parallel inside the same JVM.
18   *
19   * To enable this Run listener with maven, you will need to configure your maven-surefire-plugin with:
20   *
21   *       <plugin>
22   *       <groupId>org.apache.maven.plugins</groupId>
23   *       <artifactId>maven-surefire-plugin</artifactId>
24   *       <configuration>
25   *         <properties>
26   *           <property>
27   *             <name>listener</name>
28   *             <value>org.spf4j.junit.Spf4jRunListener</value>
29   *           </property>
30   *           <property>
31   *             <name>spf4j.junit.sampleTimeMillis</name>
32   *             <value>5</value>
33   *           </property>
34   *         </properties>
35   *       </configuration>
36   *     </plugin>
37   *
38   * and add spf4j-junit to your test classpath:
39   *
40   *   <dependency>
41   *     <groupId>org.spf4j</groupId>
42   *     <artifactId>spf4j-junit</artifactId>
43   *     <version>${spf4j.version}</version>
44   *     <scope>test</scope>
45   *   </dependency>
46   *
47   * @author zoly
48   */
49  @SuppressFBWarnings("PATH_TRAVERSAL_IN")
50  public final class Spf4jRunListener extends RunListener {
51  
52    private final Sampler sampler = new Sampler(Integer.getInteger("spf4j.junit.sampleTimeMillis", 5),
53            Integer.getInteger("spf4j.junit.dumpAfterMillis", Integer.MAX_VALUE),
54            new FastStackCollector(true));
55  
56    private final File destinationFolder = new File(System.getProperty("spf4j.junit.destinationFolder",
57            "target/junit-ssdump"));
58    {
59      if (!destinationFolder.mkdirs() && !destinationFolder.canWrite()) {
60        throw new ExceptionInInitializerError("Unable to write to " + destinationFolder);
61      }
62    }
63  
64    @Override
65    public void testFailure(final Failure failure)
66            throws IOException, InterruptedException, ExecutionException, TimeoutException {
67      sampler.stop();
68      File dumpToFile = sampler.dumpToFile(new File(destinationFolder, failure.getTestHeader() + ".ssdump2"));
69      if (dumpToFile != null) {
70        System.out.print("Profile saved to " + dumpToFile);
71      }
72    }
73  
74    @Override
75    public void testFinished(final Description description)
76            throws IOException, InterruptedException, ExecutionException, TimeoutException {
77      sampler.stop();
78      File dumpToFile = sampler.dumpToFile(new File(destinationFolder, description.getDisplayName() + ".ssdump2"));
79      if (dumpToFile != null) {
80        System.out.print("Profile saved to " + dumpToFile);
81      }
82    }
83  
84    @Override
85    public void testStarted(final Description description) {
86      sampler.start();
87    }
88  
89    @Override
90    public String toString() {
91      return "Spf4jRunListener{" + "sampler=" + sampler + '}';
92    }
93  
94  }
