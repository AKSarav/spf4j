18  package org.spf4j.io.tcp.proxy;
19  
20  import java.io.IOException;
21  import java.nio.ByteBuffer;
22  import java.nio.channels.SocketChannel;
23  
24  /**
25   *
26   * @author zoly
27   */
28  public final class TransferBuffer {
29  
30      public enum Operation { READ, WRITE };
31  
32  
33      private final ByteBuffer buffer;
34  
35  
36      private Operation lastOperation;
37  
38      private boolean isEof;
39      
40      private Runnable isDataInBufferHook;
41  
42      private Runnable isRoomInBufferHook;
43  
44      private Sniffer incomingSniffer;
45  
46  
47      public TransferBuffer(final int bufferSize) {
48          buffer = ByteBuffer.allocateDirect(bufferSize);
49          lastOperation = Operation.READ;
50          isEof = false;
51          this.isDataInBufferHook = null;
52          this.isRoomInBufferHook = null;
53      }
54  
55      public synchronized int read(final SocketChannel channel) throws IOException {
56          if (lastOperation == Operation.WRITE) {
57              buffer.compact();
58              lastOperation = Operation.READ;
59          }
60          int nrRead = channel.read(buffer);
61          if (incomingSniffer != null && (nrRead != 0)) {
62              incomingSniffer.received(buffer, nrRead);
63          }
64          if (nrRead < 0) {
65              isEof = true;
66              channel.socket().shutdownInput(); // ? is this really necessary?
67          } else if (buffer.hasRemaining()) {
68              isRoomInBufferHook.run();
69          }
70          if (buffer.position() > 0 || isEof) {
71              isDataInBufferHook.run();
72          }
73          return nrRead;
74      }
75  
76      public synchronized int write(final SocketChannel channel) throws IOException {
77          if (lastOperation == Operation.READ) {
78              buffer.flip();
79              lastOperation = Operation.WRITE;
80          }
81          int nrWritten = channel.write(buffer);
82          final boolean hasRemaining = buffer.hasRemaining();
83          if (!hasRemaining && isEof) {
84              channel.socket().shutdownOutput();
85              return nrWritten;
86          }
87          if (!isEof && buffer.position() > 0) {
88             isRoomInBufferHook.run();
89          }
90          if (hasRemaining) {
91             isDataInBufferHook.run();
92          }
93          return nrWritten;
94      }
95  
96  
97      public static int transfer(final SocketChannel in, final SocketChannel out, final ByteBuffer buffer)
98              throws IOException {
99          int read = 0;
100         //CHECKSTYLE:OFF
101         while ((in.finishConnect() && ((read = in.read(buffer)) > 0)) || (buffer.position() > 0)) {
102             //CHECKSTYLE:ON
103           buffer.flip();
104           if (!out.finishConnect()) {
105               break;
106           }
107           int writen = out.write(buffer);
108           buffer.compact();
109           if (writen <= 0) {
110             break;
111           }
112         }
113         return read;
114     }
115 
116 
117     public synchronized void setIsDataInBufferHook(final Runnable isDataInBufferHook) {
118         this.isDataInBufferHook = isDataInBufferHook;
119     }
120 
121     public synchronized void setIsRoomInBufferHook(final Runnable isRoomInBufferHook) {
122         this.isRoomInBufferHook = isRoomInBufferHook;
123     }
124 
125     public synchronized void setIncomingSniffer(final Sniffer incomingSniffer) {
126         this.incomingSniffer = incomingSniffer;
127     }
128 
129     @Override
130     public String toString() {
131         return "TransferBuffer{" + "buffer=" + buffer + ", lastOperation=" + lastOperation
132                 + ", isEof=" + isEof + ", isDataInBufferHook=" + isDataInBufferHook
133                 + ", isRoomInBufferHook=" + isRoomInBufferHook + '}';
134     }
