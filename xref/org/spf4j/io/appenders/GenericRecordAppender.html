4   import com.google.common.base.Charsets;
5   import java.io.IOException;
6   import org.apache.avro.Schema;
7   import org.apache.avro.generic.GenericDatumWriter;
8   import org.apache.avro.generic.GenericRecord;
9   import org.apache.avro.io.JsonEncoder;
10  import org.spf4j.io.AppendableOutputStream;
11  import org.spf4j.io.ObjectAppender;
12  
13  /**
14   *
15   * @author zoly
16   */
17  public final class GenericRecordAppender implements ObjectAppender<GenericRecord> {
18      
19  
20      @Override
21      public void append(final GenericRecord object, final Appendable appendTo) throws IOException {
22          try (AppendableOutputStream bos = new AppendableOutputStream(appendTo, Charsets.UTF_8)) {
23              final Schema schema = object.getSchema();
24              GenericDatumWriter<GenericRecord> writer = new GenericDatumWriter<>(schema);
25              JsonEncoder jsonEncoder = SpecificRecordAppender.EF.jsonEncoder(schema, bos);
26              writer.write(object, jsonEncoder);
27              jsonEncoder.flush();
28          }
29      }
30      
31  }
