18  package org.spf4j.zel.instr;
19  
20  import java.util.Arrays;
21  import org.spf4j.zel.vm.ExecutionContext;
22  import org.spf4j.zel.vm.Method;
23  import org.spf4j.zel.vm.Program;
24  import org.spf4j.zel.vm.SuspendedException;
25  import org.spf4j.zel.vm.ZExecutionException;
26  
27  
28  public final class CALL extends Instruction {
29  
30      private static final long serialVersionUID = 759722625722456554L;
31  
32      private CALL() {
33      }
34  
35      @Override
36      @edu.umd.cs.findbugs.annotations.SuppressWarnings("ITC_INHERITANCE_TYPE_CHECKING")
37      public int execute(final ExecutionContext context)
38              throws ZExecutionException, InterruptedException, SuspendedException {
39          Integer nrParams = (Integer) context.pop();
40          Object function = context.peekFromTop(nrParams);
41          if (function instanceof Program) {
42              final Program p = (Program) function;
43              final ExecutionContext nctx;
44              Object obj;
45              Object[] parameters;
46              switch (p.getType()) {
47                  case DETERMINISTIC:
48                      parameters = getParamsSync(context, nrParams);
49                      nctx = context.getSubProgramContext(p, parameters);
50                      obj = context.resultCache.getResult(p,  Arrays.asList(parameters), new SyncAsyncCallable(nctx));
51  
52                      break;
53                  case NONDETERMINISTIC:
54                          parameters = getParams(context, nrParams);
55                          nctx = context.getSubProgramContext(p, parameters);
56                          obj = Program.executeSyncOrAsync(nctx);
57                      break;
58                  default:
59                      throw new UnsupportedOperationException(p.getType().toString());
60              }
61              context.push(obj);
62          } else if (function instanceof Method) {
63              Object[] parameters = getParamsSync(context, nrParams);
64              try {
65                  context.push(((Method) function).invoke(context, parameters));
66              } catch (Exception ex) {
67                  throw new ZExecutionException("cannot invoke " + function, ex);
68              }
69          } else {
70              throw new ZExecutionException("cannot invoke " + function);
71          }
72          return 1;
73      }
74  
75      static Object[] getParamsSync(final ExecutionContext context, final Integer nrParams)
76              throws SuspendedException {
77          Object [] parameters;
78          try {
79              parameters = context.popSyncStackVals(nrParams);
80          } catch (SuspendedException e) {
81              context.push(nrParams); // put back param count so tat stack is identical.
82              throw e;
83          }
84          context.pop(); // extract function
85          return parameters;
86      }
87      
88      static Object[] getParams(final ExecutionContext context, final Integer nrParams)
89              throws SuspendedException {
90          Object [] parameters = context.popStackVals(nrParams);
91          context.pop();
92          return parameters;
93      }
94  
95      public static final Instruction INSTANCE = new CALL();
96  
97      @Override
98      public Object[] getParameters() {
99          return org.spf4j.base.Arrays.EMPTY_OBJ_ARRAY;
100     }
101 
102 }
