51      @Test
52      public void testCsvReadWrite() throws IOException {
53  
54          File testFile = createTestCsv();
55  
56          List<Map<String, String>> data =
57                  Csv.read(testFile, Charsets.UTF_8, new Csv.CsvHandler<List<Map<String, String>>>() {
58  
59              private boolean firstRow = true;
60  
61              private final List<Map<String, String>> result = new ArrayList<>();
62  
63              private List<String> header = new ArrayList<>();
64              private Map<String, String> row = null;
65              private int i = 0;
66  
67              @Override
68              public void startRow() {
69                  if (!firstRow) {
70                      row = new HashMap<>(header.size());
71                      i = 0;
72                  }
73              }
74  
75              @Override
76              public void element(final CharSequence elem) {
77                  if (firstRow) {
78                      header.add(elem.toString());
79                  } else {
80                      row.put(header.get(i++), elem.toString());
81                  }
82              }
83  
84              @Override
85              public void endRow() {
86                  if (firstRow) {
87                      firstRow = false;
88                  } else {
89                      result.add(row);
90                  }
91              }
92  
93              @Override
94              public List<Map<String, String>> eof() {
95                  return result;
96              }
97          });
98        Map<String, String> d0 = data.get(0);
99        Assert.assertEquals("1,3", d0.get("c"));
100       Assert.assertEquals("1", d0.get("d"));
101       Map<String, String> d1 = data.get(1);
102       Assert.assertEquals("1,3", d1.get("d"));
103       Assert.assertEquals("\"", d1.get("b"));
104       Assert.assertEquals("0\n", d1.get("c"));
105     }

135     private File createTestCsv() throws IOException {
136         File testFile = File.createTempFile("csvTest", ".csv");
137         System.out.println("test file : " + testFile);
138         try (BufferedWriter writer = new BufferedWriter(
139                 new OutputStreamWriter(new FileOutputStream(testFile), Charsets.UTF_8))) {
140             Csv.writeCsvRow(writer, "a", "b", "c", "d");
141             Csv.writeCsvRow(writer, "1.2\r", "1", "1,3", 1);
142             Csv.writeCsvRow(writer, "0", "\"", "0\n", "1,3");
143         }
144         System.out.println("test file written");
145         return testFile;
146     }
