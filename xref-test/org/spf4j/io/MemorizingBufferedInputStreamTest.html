25      private static final String TSTR =
26              "This is a super \u00EF cool, mega dupper test string for testing piping..........E";
27  
28      @Test
29      public void testSimpleStreamBuffering() throws IOException {
30          final byte[] array = Strings.toUtf8(TSTR);
31          ByteArrayInputStream bis = new ByteArrayInputStream(array);
32          MemorizingBufferedInputStream mis = new MemorizingBufferedInputStream(bis);
33          int val;
34          int i = 0;
35          while ((val = mis.read()) > 0)  {
36              System.out.print((char) val);
37              Assert.assertEquals(array[i], (byte) val);
38              i++;
39          }
40          Assert.assertEquals(i, array.length);
41      }

44      @Test
45      public void testStreamBuffering() throws IOException {
46          test(TSTR, 8, true);
47          final IntMath.XorShift32 random = new IntMath.XorShift32();
48          for (int i = 0; i < 100; i++) {
49              int nrChars = Math.abs(random.nextInt() % 100000);
50              StringBuilder sb = generateTestStr(nrChars);
51              test(sb.toString(), Math.abs(random.nextInt() % 10000), true);
52              testBuff(sb, 8192);
53              testBuff(sb, 32);
54          }
55      }
56  
57      private void testBuff(final StringBuilder sb, final int buffSize) throws IOException {
58          final byte[] utf8Bytes = Strings.toUtf8(sb.toString());
59          ByteArrayInputStream bis = new ByteArrayInputStream(utf8Bytes);
60          try (MemorizingBufferedInputStream mbis = new MemorizingBufferedInputStream(bis, buffSize, buffSize / 2,
61                  ArraySuppliers.Bytes.GL_SUPPLIER, Charsets.UTF_8)) {
62              int val = mbis.read();
63              Assert.assertEquals(val, mbis.getReadBytesFromBuffer()[0]);
64              byte [] buff = new byte [8];
65              int read = mbis.read(buff);
66              Assert.assertEquals(8, read);
67              Assert.assertTrue(Arrays.equals(buff, Arrays.copyOfRange(mbis.getReadBytesFromBuffer(), 1, 9)));
68              int result;
69              do {
70                  result = mbis.read(buff);
71              } while (result >= 0);
72              byte[] unredBytesEnd = mbis.getUnreadBytesFromBuffer();
73              Assert.assertEquals(0, unredBytesEnd.length);
74              final byte[] readBytesFromBuffer = mbis.getReadBytesFromBuffer();
75              Assert.assertTrue(Arrays.equals(readBytesFromBuffer,
76                      Arrays.copyOfRange(utf8Bytes, utf8Bytes.length - readBytesFromBuffer.length, utf8Bytes.length)));
77  
78          }
79      }
