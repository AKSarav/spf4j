29      @Test
30      @SuppressFBWarnings("SIC_INNER_SHOULD_BE_STATIC_ANON")
31      public void testBuffering() throws IOException {
32        byte [] bytes = new byte[16384];
33        for (int i = 0; i< bytes.length; i++) {
34          bytes[i] = (byte) ('a' + i % 10);
35        }
36        ByteArrayInputStream bis = new ByteArrayInputStream(bytes) {
37          private boolean first = true;
38          @Override
39          public synchronized int read(byte[] b, int off, int len) {
40            if (first && len > 260) {
41              first = false;
42              return super.read(b, off, 260);
43            } else {
44              return super.read(b, off, len);
45            }
46          }
47  
48        };
49        try (MemorizingBufferedInputStream mbis = new MemorizingBufferedInputStream(bis)) {
50          StringBuilder sb = new StringBuilder();
51          PushbackInputStreamEx pushbackInputStreamEx = new PushbackInputStreamEx(mbis);
52          int read = pushbackInputStreamEx.read();
53          pushbackInputStreamEx.unread(read);
54          CharStreams.copy(new InputStreamReader(pushbackInputStreamEx, StandardCharsets.UTF_8), sb);
55          Assert.assertEquals(16384, sb.length());
56          Assert.assertEquals(new String(bytes, StandardCharsets.US_ASCII), sb.toString());
57        }
58      }

64      @Test
65      public void testSimpleStreamBuffering() throws IOException {
66          final byte[] array = Strings.toUtf8(TSTR);
67          ByteArrayInputStream bis = new ByteArrayInputStream(array);
68          MemorizingBufferedInputStream mis = new MemorizingBufferedInputStream(bis);
69          int val;
70          int i = 0;
71          while ((val = mis.read()) > 0)  {
72              System.out.print((char) val);
73              Assert.assertEquals(array[i], (byte) val);
74              i++;
75          }
76          Assert.assertEquals(i, array.length);
77      }

98      public void testStreamBuffering() throws IOException {
99          test(TSTR, 8, true);
100         final IntMath.XorShift32 random = new IntMath.XorShift32();
101         for (int i = 0; i < 100; i++) {
102             int nrChars = Math.abs(random.nextInt() % 100000);
103             StringBuilder sb = generateTestStr(nrChars);
104             test(sb.toString(), Math.abs(random.nextInt() % 10000), true);
105             testBuff(sb, 8192);
106             testBuff(sb, 32);

110     @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
111     private void testBuff(final StringBuilder sb, final int buffSize) throws IOException {
112         final byte[] utf8Bytes = Strings.toUtf8(sb.toString());
113         ByteArrayInputStream bis = new ByteArrayInputStream(utf8Bytes);
114         try (MemorizingBufferedInputStream mbis = new MemorizingBufferedInputStream(bis, buffSize, buffSize / 2,
115                 ArraySuppliers.Bytes.GL_SUPPLIER, Charsets.UTF_8)) {
116             int val = mbis.read();
117             Assert.assertEquals(val, mbis.getReadBytesFromBuffer()[0]);
118             byte [] buff = new byte [8];
119             int read = mbis.read(buff);
120             Assert.assertEquals(8, read);
121             Assert.assertTrue(Arrays.equals(buff, Arrays.copyOfRange(mbis.getReadBytesFromBuffer(), 1, 9)));
122             int result;
123             do {
124                 result = mbis.read(buff);
125             } while (result >= 0);
126             byte[] unredBytesEnd = mbis.getUnreadBytesFromBuffer();
127             Assert.assertEquals(0, unredBytesEnd.length);
128             final byte[] readBytesFromBuffer = mbis.getReadBytesFromBuffer();
129             Assert.assertTrue(Arrays.equals(readBytesFromBuffer,
130                     Arrays.copyOfRange(utf8Bytes, utf8Bytes.length - readBytesFromBuffer.length, utf8Bytes.length)));
131 
132         }
