18  public class CompressTest {
19  
20    @Test
21    public void testZip() throws IOException {
22      File file = File.createTempFile(".bla", ".tmp");
23      String testStr = "skjfghskjdhgfjgishfgksjhgjkhdskghsfdkjhg";
24      Files.write(testStr, file, Charsets.UTF_8);
25      Path zip = Compress.zip(file.toPath());
26      Assert.assertThat(zip.getFileName().toString(), Matchers.endsWith(".zip"));
27      Assert.assertTrue(java.nio.file.Files.exists(zip));
28      File tmpFir = Files.createTempDir();
29      List<Path> unzip = Compress.unzip(zip, tmpFir.toPath());
30      Assert.assertEquals(1, unzip.size());
31      Assert.assertEquals(testStr, Files.toString(unzip.get(0).toFile(), Charsets.UTF_8));
32    }
