6   import java.lang.management.ManagementFactory;
7   import java.util.Arrays;
8   import junit.framework.Assert;
9   import org.junit.Test;
10  import org.spf4j.base.AbstractRunnable;
11  
12  /**
13   *
14   * @author zoly
15   */
16  public final class FileBasedLockTest {
17  
18      private static final String LOCK_FILE = org.spf4j.base.Runtime.TMP_FOLDER + File.separatorChar
19              + "test.lock";
20      private volatile boolean isOtherRunning;
21  
22      @Test(timeout = 60000)
23      public void test() throws IOException, InterruptedException {
24  
25          final String classPath = ManagementFactory.getRuntimeMXBean().getClassPath();
26          final String jvmPath =
27                  System.getProperties().getProperty("java.home")
28                  + File.separatorChar + "bin" + File.separatorChar + "java";
29  
30  
31          Thread extProcThread = new Thread(new AbstractRunnable() {
32              @Override
33              public void doRun() throws IOException, InterruptedException {
34                  String[] command = new String[]{jvmPath, "-cp", classPath, FileBasedLockTest.class.getName()};
35                  System.out.println("Running " + Arrays.toString(command));
36                  Process proc = Runtime.getRuntime().exec(command);
37                  int result = proc.waitFor();
38                  isOtherRunning = !(result == 0);
39              }
40          });
41          FileBasedLock lock = new FileBasedLock(new File(LOCK_FILE));
42          lock.lock();
43          try {
44              extProcThread.start();
45              isOtherRunning = true;
46              for (int i = 1; i < 10000; i++) {
47                  if (!isOtherRunning) {
48                      Assert.fail("The Other process should be running, lock is not working");
49                  }
50                  Thread.sleep(1);
51              }
52          } finally {
53              lock.unlock();
54          }
55          extProcThread.join();
56          System.out.println("Lock test successful");
57  
58      }
59  
60      public static void main(final String[] args) throws FileNotFoundException, InterruptedException {
61          FileBasedLock lock = new FileBasedLock(new File(LOCK_FILE));
62          lock.lock();
63          try {
64              Thread.sleep(100);
65          } finally {
66              lock.unlock();
67          }
68          System.exit(0);
69      }
70  }
