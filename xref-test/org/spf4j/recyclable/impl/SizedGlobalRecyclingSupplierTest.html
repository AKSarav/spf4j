4   import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
5   import org.junit.Assert;
6   import org.junit.Test;
7   import org.spf4j.base.ReferenceType;
8   import org.spf4j.recyclable.SizedRecyclingSupplier;
9   
10  /**
11   *
12   * @author zoly
13   */
14  @SuppressFBWarnings("SIC_INNER_SHOULD_BE_STATIC_ANON")
15  public class SizedGlobalRecyclingSupplierTest {
16  
17      @Test
18      @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
19      public void testSupplier() {
20          SizedRecyclingSupplier<byte[]> supplier = new Powerof2SizedGlobalRecyclingSupplier<>(
21                  new SizedRecyclingSupplier.Factory<byte[]>() {
22  
23              @Override
24              @SuppressFBWarnings("SUA_SUSPICIOUS_UNINITIALIZED_ARRAY")
25              public byte[] create(final int size) {
26                  return new byte[size];
27              }
28  
29              @Override
30              public int size(final byte[] object) {
31                  return object.length;
32              }
33          }, ReferenceType.SOFT);
34  
35          byte[] get = supplier.get(8000);
36          supplier.recycle(get);
37          byte [] get2 = supplier.get(8000);
38          supplier.recycle(get2);
39          Assert.assertTrue(get == get2);
40  
41          get = supplier.get(8000);
42          get2 = supplier.get(8000);
43          Assert.assertTrue(get != get2);
44          supplier.recycle(get);
45          supplier.recycle(get2);
46  
47          get = supplier.get(4000);
48          Assert.assertEquals(4096, get.length);
49          supplier.recycle(get);
50          get2 = supplier.get(4000);
51          Assert.assertEquals(4096, get2.length);
52          supplier.recycle(get2);
53          Assert.assertTrue(get == get2);
54  
55      }
