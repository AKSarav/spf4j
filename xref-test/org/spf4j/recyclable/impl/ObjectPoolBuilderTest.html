109     private volatile boolean isDeadlock = false;
110     
111     private Thread startDeadlockMonitor(final long deadlockTimeout) {
112         isDeadlock = false;
113         Thread monitor = new Thread(new Runnable() {
114 
115             @Override
116             public void run() {
117                 try {
118                     Thread.sleep(deadlockTimeout);
119                     ThreadMXBean threadMX = ManagementFactory.getThreadMXBean();
120                     System.err.println(Arrays.toString(threadMX.dumpAllThreads(true, true)));
121                     isDeadlock = true;
122                 } catch (InterruptedException ex) {
123                     // terminating monitor
124                     return;
125                 }
126             }
127         });
128         monitor.start();
129         return monitor;
130     }
131    private void runTest(final RecyclingSupplier<ExpensiveTestObject> pool,
132             final long sleepBetweenSubmit, final long deadlockTimeout) throws InterruptedException, ExecutionException {
133         Thread monitor = startDeadlockMonitor(deadlockTimeout);
134         ExecutorService execService = Executors.newFixedThreadPool(10);
135         BlockingQueue<Future<Integer>> completionQueue = new LinkedBlockingDeque<Future<Integer>>();
136         RetryExecutor<Integer> exec
137                 = new RetryExecutor<Integer>(execService, 8, 16, 5000, Callables.DEFAULT_EXCEPTION_RETRY,
138                  completionQueue);
139         int nrTests = 1000;
140         for (int i = 0; i < nrTests; i++) {
141             exec.submit(new TestCallable(pool, i));
142             Thread.sleep(sleepBetweenSubmit);
143         }
144         for (int i = 0; i < nrTests; i++) {
145             System.out.println("Task " + completionQueue.take().get() + " finished ");
146         }
147         monitor.interrupt();
148         monitor.join();
149         Thread.sleep(100);
150         Assert.assertEquals(0, completionQueue.size());
151         if (isDeadlock) {
152             Assert.fail("deadlock detected");
153         }
154     }
