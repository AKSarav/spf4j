5   package org.spf4j.stackmonitor;
6   
7   import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
8   import org.junit.Assert;
9   import org.junit.Test;
10  
11  /**
12   *
13   * @author zoly
14   */
15  public final class SampleNodeTest {
16  
17      public SampleNodeTest() {
18      }
19  
20      @Test
21      @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
22      public void testSampleNode() {
23  
24          System.out.println("sample");
25          StackTraceElement[] st1 = new StackTraceElement[3];
26          st1[0] = new StackTraceElement("C1", "m1", "C1.java", 10);
27          st1[1] = new StackTraceElement("C1", "m2", "C1.java", 11);
28          st1[2] = new StackTraceElement("C1", "m3", "C1.java", 12);
29          SampleNode node1 = SampleNode.createSampleNode(st1);
30          SampleNode node2 = new SampleNode(st1, st1.length - 1);
31          System.out.println(node1);
32          System.out.println(node2);
33  
34          Assert.assertEquals(4, node1.getNrNodes());
35          Assert.assertEquals(4, node2.getNrNodes());
36  
37  
38          StackTraceElement[] st2 = new StackTraceElement[1];
39          st2[0] = new StackTraceElement("C1", "m1", "C1.java", 10);
40          SampleNode.addToSampleNode(node1, st2);
41          node2.addSample(st2, st2.length - 1);
42          System.out.println(node1);
43          System.out.println(node2);
44          Assert.assertEquals(5, node1.getNrNodes());
45          Assert.assertEquals(5, node2.getNrNodes());
46  
47          StackTraceElement[] st3 = new StackTraceElement[3];
48          st3[0] = new StackTraceElement("C2", "m1", "C2.java", 10);
49          st3[1] = new StackTraceElement("C2", "m2", "C2.java", 11);
50          st3[2] = new StackTraceElement("C2", "m3", "C2.java", 12);
51          SampleNode.addToSampleNode(node1, st3);
52          node2.addSample(st3, st3.length - 1);
53          System.out.println(node1);
54          System.out.println(node2);
55  
56          StackTraceElement[] st4 = new StackTraceElement[3];
57          st4[0] = new StackTraceElement("C1", "m1", "C1.java", 10);
58          st4[1] = new StackTraceElement("C1", "m2", "C1.java", 11);
59          st4[2] = new StackTraceElement("C1", "m4", "C1.java", 14);
60          SampleNode.addToSampleNode(node1, st4);
61          node2.addSample(st4, st4.length - 1);
62  
63          SampleNode.addToSampleNode(node1, st1);
64          node2.addSample(st1, st1.length - 1);
65  
66          System.out.println("n1 = " + node1);
67          System.out.println("n2 = " + node2);
68          Assert.assertEquals(node1.toString(), node2.toString());
69  
70          SampleNode agg = SampleNode.aggregate(node1, node2);
71          System.out.println("n1 + n2 = " + agg);
72          Assert.assertEquals(node1.getSampleCount() + node2.getSampleCount(), agg.getSampleCount());
73          final Method method = new Method("C1", "m3");
74          Assert.assertEquals(node1.getSubNodes().get(method).getSampleCount()
75                  + node2.getSubNodes().get(method).getSampleCount(),
76                  agg.getSubNodes().get(method).getSampleCount());
77  
78      }
79  }
