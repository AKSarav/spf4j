15  public final class ScannerTest {

17  
18      public static class A {
19          Object getValue() {
20              return new Object();
21          }
22      }
23  
24      public static class B extends A {
25          @Override
26          String getValue() {
27              return "B";
28          }
29      }
30  
31      @Test
32      public void testSomeMethod() throws NoSuchMethodException {
33         String desc  = Type.getMethodDescriptor(B.class.getDeclaredMethod("getValue"));
34         System.out.println(desc + new A().getValue());
35         String desc2  = Type.getMethodDescriptor(A.class.getDeclaredMethod("getValue"));
36         System.out.println(desc2 + ((A) new B()).getValue());
37         // read a System property
38         System.getProperty("some.property", "default value");
39         Integer.getInteger("someInt.value", 3);
40         Long.getLong("someLong.value", 5);
41      }
42  
43  
44      @Test
45      public void testScan() throws NoSuchMethodException, IOException {
46  
47          List<Invocation> findUsages = Scanner.findUsages(ScannerTest.class,
48                  ImmutableSet.of(A.class.getDeclaredMethod("getValue")));
49          System.out.println(findUsages);
50          final ImmutableSet<Method> lookFor = ImmutableSet.of(System.class.getDeclaredMethod("getProperty", String.class),
51                  System.class.getDeclaredMethod("getProperty", String.class, String.class),
52                  Integer.class.getDeclaredMethod("getInteger", String.class),
53                  Integer.class.getDeclaredMethod("getInteger", String.class, int.class),
54                  Integer.class.getDeclaredMethod("getInteger", String.class, Integer.class),
55                  Long.class.getDeclaredMethod("getLong", String.class),
56                  Long.class.getDeclaredMethod("getLong", String.class, Long.class),
57                  Long.class.getDeclaredMethod("getLong", String.class, long.class),
58                  Boolean.class.getDeclaredMethod("getBoolean", String.class));
59          List<Invocation> findUsages2 = Scanner.findUsages(ScannerTest.class, lookFor);
60          System.out.println("Scan 1 = " + findUsages2);
61          List<Invocation> findUsages3 = Scanner.findUsages(ScannerTest.class.getPackage().getName(), lookFor);
62          System.out.println("Scan 2 = " + findUsages3);
63  
64      }
