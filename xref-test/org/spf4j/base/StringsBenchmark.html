14   * @author zoly

16  @State(Scope.Benchmark)
17  @Fork(2)
18  @Threads(value = 4)
19  public class StringsBenchmark {
20  
21      public static final String TEST_STRING;
22  
23      static {
24          IntMath.XorShift32 rnd = new IntMath.XorShift32();
25          StringBuilder builder = new StringBuilder(2200);
26          for (int i = 0; i < 2200; i++) {
27              builder.append('A' + Math.abs(rnd.nextInt()) % 22);
28          }
29          builder.append("cw==");
30          TEST_STRING = builder.toString();
31      }
32  
33      @Benchmark
34      public final byte [] optimizedSubStringDecode() throws UnsupportedEncodingException {
35          return BaseEncoding.base64().decode(Strings.subSequence(TEST_STRING, 10, TEST_STRING.length()));
36      }
37  
38      @Benchmark
39      public final byte [] subStringDecode() {
40          return BaseEncoding.base64().decode(TEST_STRING.subSequence(10, TEST_STRING.length()));
41      }
