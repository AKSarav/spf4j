13  public final class RTreeTest {

15      /**
16       * Test of search method, of class RTree.
17       */
18      @Test
19      public void testSearch() {
20          RTree<String> rectangles = new RTree<String>();
21          rectangles.insert(new float [] {1, 1}, "Point 1,1");
22          rectangles.insert(new float [] {0.5f, 0.5f}, new float [] {1, 1}, "Rectangle 0.5, 0.5, 1, 1");
23          List<String> result = rectangles.search(new float [] {0, 0}, new float [] {2, 2});
24          Assert.assertEquals(2, result.size());
25          List<String> result2 = rectangles.search(new float [] {1.5f, 1.5f}, new float [] {2, 2});
26          Assert.assertEquals(1, result2.size());
27  
28          RTree<String> ranges = new RTree<String>(1);
29          ranges.insert(new float [] {1}, new float [] {1}, "range 1,2");
30          ranges.insert(new float [] {2}, new float [] {2}, "range 2,4");
31          Assert.assertEquals(0, ranges.search(new float [] {10}, new float [] {0}).size());
32      }
