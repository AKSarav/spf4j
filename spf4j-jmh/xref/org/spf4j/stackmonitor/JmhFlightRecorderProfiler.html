22  public final class JmhFlightRecorderProfiler implements ExternalProfiler {
23  
24      private static final String DUMP_FOLDER = System.getProperty("jmh.stack.profiles", org.spf4j.base.Runtime.USER_DIR);
25  
26      private static final String DEFAULT_OPTIONS = System.getProperty("jmh.fr.options",
27              "defaultrecording=true,settings=profile");
28  
29  
30  
31      @Override
32      public Collection<String> addJVMInvokeOptions(final BenchmarkParams params) {
33          return Collections.emptyList();
34      }
35  
36      private volatile String dumpFile;
37  
38      private static volatile String benchmarkName;
39  
40      public static String benchmarkName() {
41          return benchmarkName;
42      }
43  
44  
45      /**
46       * See:
47       * http://docs.oracle.com/cd/E15289_01/doc.40/e15070/usingjfr.htm
48       * and
49       * http://docs.oracle.com/cd/E15289_01/doc.40/e15070/config_rec_data.htm
50       * @param params
51       * @return
52       */
53      @Override
54      @SuppressFBWarnings("ST_WRITE_TO_STATIC_FROM_INSTANCE_METHOD")
55      public Collection<String> addJVMOptions(final BenchmarkParams params) {
56          final String id = params.id();
57          benchmarkName = id;
58          dumpFile = DUMP_FOLDER + '/' + id + ".jfr";
59          String flightRecorderOptions = DEFAULT_OPTIONS + ",dumponexit=true,dumponexitpath=" + dumpFile;
60          return Arrays.asList(
61                  "-XX:+FlightRecorder",
62                  "-XX:FlightRecorderOptions=" + flightRecorderOptions);
63      }
64  
65      @Override
66      public void beforeTrial(final BenchmarkParams benchmarkParams) {
67          final List<String> inputArguments = ManagementFactory.getRuntimeMXBean().getInputArguments();
68          if (!inputArguments
69                  .contains("-XX:+UnlockCommercialFeatures")) {
70              throw new RuntimeException("-XX:+UnlockCommercialFeatures must pre present in the JVM options,"
71                      + " current options are: " + inputArguments);
72          }
73      }
74  
75  
76      @Override
77      public boolean allowPrintOut() {
78          return true;
79      }
80  
81      @Override
82      public boolean allowPrintErr() {
83          return false;
84      }
85  
86  
87      @Override
88      public String getDescription() {
89          return "Java Flight Recording profiler runs for every benchmark.";
90      }
91  
92      @Override
93      public Collection<? extends Result> afterTrial(final BenchmarkResult bp, final long l,
94              final File file, final File file1) {
95          NoResult r = new NoResult("Profile saved to " + dumpFile + ", results: " + bp
96              + ", stdOutFile = " + file + ", stdErrFile = " + file1);
97          return Collections.singleton(r);
98      }
99  
100     private static final class NoResult extends Result<NoResult> {
101         private static final long serialVersionUID = 1L;
102 
103         private final String output;
104 
105         public NoResult(final String output) {
106             super(ResultRole.SECONDARY, "JFR", of(Double.NaN), "N/A", AggregationPolicy.SUM);
107             this.output = output;
108         }
109 
110         @Override
111         protected Aggregator<NoResult> getThreadAggregator() {
112             return new NoResultAggregator();
113         }
114 
115         @Override
116         protected Aggregator<NoResult> getIterationAggregator() {
117             return new NoResultAggregator();
118         }
119 
120         private static class NoResultAggregator implements Aggregator<NoResult> {
121 
122             @Override
123             public NoResult aggregate(final Collection<NoResult> results) {
124                 StringBuilder agg = new StringBuilder();
125                 for (NoResult r : results) {
126                     agg.append(r.output);
127                 }
128                 return new NoResult(agg.toString());
129             }
130         }
131     }
