26                  new long [] [] {{1L, 2L, 3L}, {10L, 20L, 30L}},
27                  10, 3);
28          Assert.assertEquals(4, fillGaps.getFirst().length);
29          Assert.assertEquals(4, fillGaps.getSecond().length);
30          Assert.assertEquals(10L, fillGaps.getFirst()[1]);
31          Assert.assertEquals(20L, fillGaps.getFirst()[2]);
32          Assert.assertEquals(Double.NaN, fillGaps.getSecond()[1][2]);
