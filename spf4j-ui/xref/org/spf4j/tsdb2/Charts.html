25     public static boolean canGenerateMinMaxAvgCount(final TableDef info) {
26         int found = 0;
27         for (ColumnDef colDef : info.getColumns()) {
28             switch (colDef.getName()) {
29                 case "min":
30                 case "max":
31                 case "total":
32                 case "count":
33                     found++;
34                     break;
35                 default:
36             }
37         }
38         return found >= 4;
39      }

55      public static boolean canGenerateHeatChart(final TableDef info) {
56          for (ColumnDef colDef : info.getColumns()) {
57              if (colDef.name.startsWith("Q") && colDef.name.contains("_")) {
58                  return true;
59              }
60          }
61          return false;
62      }

71     public static JFreeChart createHeatJFreeChart(final TimeSeries data, final TableDef info) {
72          Pair<long[], double[][]> mData = fillGaps(data.getTimeStamps(), data.getValues(),
73                  info.getSampleTime(), info.getColumns().size());
74          ColumnDef columnDef = TSDBQuery.getColumnDef(info, "total");
75          return org.spf4j.perf.impl.chart.Charts.createHeatJFreeChart(TSDBQuery.getColumnNames(info),
76                  mData.getSecond(), data.getTimeStamps()[0], info.getSampleTime(),
77                  columnDef.getUnitOfMeasurement(), "Measurements distribution for "
78                  + info.getName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j");
79      }

81      public static JFreeChart createMinMaxAvgJFreeChart(final TimeSeries data, final TableDef info) {
82          long[][] vals = data.getValues();
83          double[] min = Arrays.getColumnAsDoubles(vals, TSDBQuery.getColumnIndex(info, "min"));
84          double[] max = Arrays.getColumnAsDoubles(vals, TSDBQuery.getColumnIndex(info, "max"));
85          final int totalColumnIndex = TSDBQuery.getColumnIndex(info, "total");
86          double[] total = Arrays.getColumnAsDoubles(vals, totalColumnIndex);
87          double[] count = Arrays.getColumnAsDoubles(vals, TSDBQuery.getColumnIndex(info, "count"));
88          for (int i = 0; i < count.length; i++) {
89              if (count[i] == 0) {
90                  min[i] = 0;
91                  max[i] = 0;
92              }
93          }
94          long[] timestamps = data.getTimeStamps();
95          return org.spf4j.perf.impl.chart.Charts.createTimeSeriesJFreeChart("Min,Max,Avg chart for "
96                  + info.getName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j", timestamps,
97                  new String[]{"min", "max", "avg"},
98                  TSDBQuery.getColumnDef(info, "total").getUnitOfMeasurement(),
99                  new double[][]{min, max, Arrays.divide(total, count)});
100     }

125     public static List<JFreeChart> createJFreeCharts(final TimeSeries data, final TableDef info) {
126         long[][] vals = data.getValues();
127         List<JFreeChart> result = new ArrayList<>();
128         Map<String, Pair<List<String>, List<double[]>>> measurementsByUom = new HashMap<>();
129         //String[] columnMetaData = TSDBQuery.getColumnUnitsOfMeasurement(info);
130         int i = 0;
131         for (ColumnDef colDef : info.getColumns()) {
132             String uom = colDef.getUnitOfMeasurement();
133             Pair<List<String>, List<double[]>> meas = measurementsByUom.get(uom);
134             if (meas == null) {
135                 meas = Pair.of((List<String>) new ArrayList<String>(), (List<double[]>) new ArrayList<double[]>());
136                 measurementsByUom.put(uom, meas);
137             }
138             meas.getFirst().add(colDef.getName());
139             meas.getSecond().add(Arrays.getColumnAsDoubles(vals, i));
140             i++;
141         }
142         long[] timestamps = data.getTimeStamps();
143         for (Map.Entry<String, Pair<List<String>, List<double[]>>> entry : measurementsByUom.entrySet()) {
144             Pair<List<String>, List<double[]>> p = entry.getValue();
145             final List<String> measurementNames = p.getFirst();
146             final List<double[]> measurements = p.getSecond();
147             result.add(org.spf4j.perf.impl.chart.Charts.createTimeSeriesJFreeChart("chart for "
148                     + info.getName() + ", sampleTime " + info.getSampleTime()
149                     + " ms, generated by spf4j", timestamps,
150                     measurementNames.toArray(new String[measurementNames.size()]), entry.getKey(),
151                     measurements.toArray(new double[measurements.size()][])));
152         }
153         return result;
154     }
