60  /**
61   * Yet another time series database. Why? because all the other ts databases had various constraints that restrict the
62   * functionality I can add to spf4j.
63   *
64   * Initial Features:
65   *
66   * 1. measurements can be added dynamically anytime to a database. 2. long measurement names. 3. the stored interval is
67   * not known from the beginning. 4. implementation biased towards write performance.
68   *
69   * Format:
70   *
71   * Header: TSDB[version:int][metadata:bytes] Table of Contents: firstTableInfoPtr, lastTableInfoPtr TableInfo:
72   * DataFragment: ...... ...... TableInfo: DataFragment: EOF
73   *
74   * @deprecated please use org.spf4j.tsdb2
75   * @author zoly
76   */

102   public TimeSeriesDatabase(final File pathToDatabaseFile, final boolean isWrite, final byte... metaData)
103           throws IOException {
104     file = new RandomAccessFile(pathToDatabaseFile, isWrite ? "rw" : "r");
105     // uniques per process string for sync purposes.
106     this.path = INTERNER.intern(pathToDatabaseFile.getPath());
107     tables = new ConcurrentHashMap<>();
108     writeDataFragments = new HashMap<>();
109     // read or create header
110     synchronized (path) {
111       this.ch = file.getChannel();
112       FileLock lock;
113       if (isWrite) {
114         lock = ch.lock();
115       } else {
116         lock = ch.lock(0, Long.MAX_VALUE, true);
117       }
118       try {
119         if (file.length() == 0) {
120           this.header = new Header(VERSION, metaData);
121           this.header.writeTo(file);
122           this.toc = new TableOfContents(file.getFilePointer());
123           this.toc.writeTo(file);
124         } else {
125           this.header = new Header(file);
126           this.toc = new TableOfContents(file);
127         }
128       } catch (IOException | RuntimeException e) {
129         try {
130           lock.release();
131           throw e;
132         } catch (IOException ex) {
133           ex.addSuppressed(e);
134           throw ex;
135         }
136       }
137       lock.release();
138       lock = ch.lock(0, Long.MAX_VALUE, true);
139       try {
140         readTableInfos();
141       } catch (IOException | RuntimeException e) {
142         try {
143           lock.release();
144           throw e;
145         } catch (IOException ex) {
146           ex.addSuppressed(e);
147           throw ex;
148         }
149       }
150       lock.release();
151     }
152   }

154   public void reReadTableInfos() throws IOException {
155     synchronized (path) {
156       FileLock lock = ch.lock(0, Long.MAX_VALUE, true);
157       try {
158         toc = new TableOfContents(file, toc.getLocation()); // reread toc
159         readTableInfos();
160       } catch (IOException | RuntimeException e) {
161         try {
162           lock.release();
163           throw e;
164         } catch (IOException ex) {
165           ex.addSuppressed(e);
166           throw ex;
167         }
168       }
169       lock.release();
170     }
171   }

173   private void readTableInfos() throws IOException {
174     final long firstColumnInfo = toc.getFirstTableInfoPtr();
175     if (firstColumnInfo > 0) {
176       file.seek(firstColumnInfo);
177       TSTable colInfo = new TSTable(file);
178       tables.put(colInfo.getTableName(), colInfo);
179       lastTableInfo = colInfo;
180       while (colInfo.getNextTSTable() > 0) {
181         file.seek(colInfo.getNextTSTable());
182         colInfo = new TSTable(file);
183         tables.put(colInfo.getTableName(), colInfo);
184         lastTableInfo = colInfo;
185       }
186     }
187   }
188 
189   private void readLastTableInfo() throws IOException {
190     toc = new TableOfContents(file, toc.getLocation()); // reread toc
191     if (toc.getLastTableInfoPtr() == 0) {
192       return;
193     }
194     lastTableInfo = new TSTable(file, toc.getLastTableInfoPtr()); // update last table info
195   }

222   public void addTSTable(final String tableName,
223           final byte[] tableMetaData, final int sampleTime, final String[] columnNames,
224           final byte[][] columnMetaData) throws IOException {
225     synchronized (path) {
226       if (hasTSTable(tableName)) {
227         throw new IllegalArgumentException("group already exists " + tableName);
228       }
229       flush();
230       FileLock lock = ch.lock();
231       TSTable colInfo;
232       try {
233         readLastTableInfo();
234         //write column information at the end of the file.
235         file.seek(file.length());
236         colInfo = new TSTable(tableName, tableMetaData, columnNames,
237                 columnMetaData, sampleTime, file.getFilePointer());
238         colInfo.writeTo(file);
239         //update refferences to this new TableInfo.
240         if (lastTableInfo != null) {
241           lastTableInfo.setNextColumnInfo(colInfo.getLocation(), file);
242         } else {
243           toc.setFirstTableInfo(colInfo.getLocation(), file);
244         }
245         toc.setLastTableInfo(colInfo.getLocation(), file);
246       } catch (IOException | RuntimeException e) {
247         try {
248           lock.release();
249           throw e;
250         } catch (IOException ex) {
251           ex.addSuppressed(e);
252           throw ex;
253         }
254       }
255       lock.release();
256 
257       lastTableInfo = colInfo;
258       tables.put(tableName, colInfo);
259     }
260   }

262   public void write(final long time, final String tableName, final long[] values) {
263     if (!hasTSTable(tableName)) {
264       throw new IllegalArgumentException("Unknown table name" + tableName);
265     }
266     synchronized (writeDataFragments) {
267       DataFragment writeDataFragment = writeDataFragments.get(tableName);
268       if (writeDataFragment == null) {
269         writeDataFragment = new DataFragment(time);
270         writeDataFragments.put(tableName, writeDataFragment);
271       }
272       writeDataFragment.addData(time, values);
273     }
274   }

276   public void flush() throws IOException {
277     synchronized (path) {
278       List<Map.Entry<String, DataFragment>> lwriteDataFragments;
279       synchronized (writeDataFragments) {
280         if (writeDataFragments.isEmpty()) {
281           return;
282         }
283         lwriteDataFragments = new ArrayList<>(writeDataFragments.entrySet());
284         writeDataFragments.clear();
285       }
286       FileLock lock = ch.lock();
287       try {
288         for (Map.Entry<String, DataFragment> entry : lwriteDataFragments) {
289           DataFragment writeDataFragment = entry.getValue();
290           String groupName = entry.getKey();
291           file.seek(file.length());
292           writeDataFragment.setLocation(file.getFilePointer());
293           writeDataFragment.writeTo(file);
294           TSTable colInfo = tables.get(groupName);
295           colInfo = new TSTable(file, colInfo.getLocation()); // reread colInfo
296           tables.put(groupName, colInfo); // update colInfo
297           final long lastDataFragment = colInfo.getLastDataFragment();
298           final long location = writeDataFragment.getLocation();
299           if (lastDataFragment != 0) {
300             DataFragment.setNextDataFragment(lastDataFragment, location, file);
301           } else {
302             colInfo.setFirstDataFragment(location, file);
303           }
304           colInfo.setLastDataFragment(location, file);
305         }
306         sync();
307       } catch (IOException | RuntimeException e) {
308         try {
309           lock.release();
310           throw e;
311         } catch (IOException ex) {
312           ex.addSuppressed(e);
313           throw ex;
314         }
315       }
316       lock.release();
317     }
318   }

382   /**
383    * Read measurements from table.
384    *
385    * @param tableName
386    * @param startTime start time including
387    * @param endTime end time including
388    * @return
389    * @throws IOException
390    */
391   private TimeSeries read(
392           final long startTime, final long endTime,
393           final long startAtFragment, final long endAtFragment,
394           final boolean skipFirst)
395           throws IOException {
396     synchronized (path) {
397       TLongArrayList timeStamps = new TLongArrayList();
398       List<long[]> data = new ArrayList<>();
399       if (startAtFragment > 0) {
400         FileLock lock = ch.lock(0, Long.MAX_VALUE, true);
401         try {
402           DataFragment frag;
403           long nextFragmentLocation = startAtFragment;
404           boolean last = false;
405           boolean psFirst = skipFirst;
406           do {
407             if (nextFragmentLocation == endAtFragment) {
408               last = true;
409             }
410             file.seek(nextFragmentLocation);
411             frag = new DataFragment(file);
412             if (psFirst) {
413               psFirst = false;
414             } else {
415               long fragStartTime = frag.getStartTimeMillis();
416               if (fragStartTime >= startTime) {
417                 TIntArrayList fragTimestamps = frag.getTimestamps();
418                 int nr = 0;
419                 for (int i = 0; i < fragTimestamps.size(); i++) {
420                   long ts = fragStartTime + fragTimestamps.get(i);
421                   if (ts <= endTime) {
422                     timeStamps.add(ts);
423                     nr++;
424                   } else {
425                     break;
426                   }
427                 }
428                 int i = 0;
429                 for (long[] d : frag.getData()) {
430                   if (i < nr) {
431                     data.add(d);
432                   } else {
433                     break;
434                   }
435                   nr++;
436                 }
437                 if (fragTimestamps.size() > nr) {
438                   break;
439                 }
440               }
441             }
442             nextFragmentLocation = frag.getNextDataFragment();
443           } while (nextFragmentLocation > 0 && !last);
444         } catch (IOException | RuntimeException e) {
445           try {
446             lock.release();
447             throw e;
448           } catch (IOException ex) {
449             ex.addSuppressed(e);
450             throw ex;
451           }
452         }
453         lock.release();
454       }
455       return new TimeSeries(timeStamps.toArray(), data.toArray(new long[data.size()][]));
456     }
457   }

514     final int totalColumnIndex = info.getColumnIndex("total");

517             new String(info.getColumnMetaData()[totalColumnIndex], Charsets.UTF_8), "Measurements distribution for "
518             + info.getTableName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j");

521   public static JFreeChart createMinMaxAvgJFreeChart(final TimeSeries data, final TSTable info) {
522     long[][] vals = data.getValues();
523     double[] min = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("min"));
524     double[] max = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("max"));
525     final int totalColumnIndex = info.getColumnIndex("total");
526     double[] total = Arrays.getColumnAsDoubles(vals, totalColumnIndex);
527     double[] count = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("count"));
528     for (int i = 0; i < count.length; i++) {
529       if (count[i] == 0) {
530         min[i] = 0;
531         max[i] = 0;
532       }
533     }
534     long[] timestamps = data.getTimeStamps();
535     return Charts.createTimeSeriesJFreeChart("Min,Max,Avg chart for "
536             + info.getTableName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j", timestamps,
537             new String[]{"min", "max", "avg"}, new String(info.getColumnMetaData()[totalColumnIndex],
538                     Charsets.UTF_8), new double[][]{min, max, Arrays.divide(total, count)});
539   }

550   public static List<JFreeChart> createJFreeCharts(final TimeSeries data, final TSTable info) {
551     long[][] vals = data.getValues();
552     List<JFreeChart> result = new ArrayList<>();
553     Map<String, Pair<List<String>, List<double[]>>> measurementsByUom = new HashMap<>();
554     String[] columnMetaData = info.getColumnMetaDataAsStrings();
555     for (int i = 0; i < info.getColumnNumber(); i++) {
556       String uom = columnMetaData[i];
557       Pair<List<String>, List<double[]>> meas = measurementsByUom.get(uom);
558       if (meas == null) {
559         meas = Pair.of((List<String>) new ArrayList<String>(), (List<double[]>) new ArrayList<double[]>());
560         measurementsByUom.put(uom, meas);
561       }
562       meas.getFirst().add(info.getColumnName(i));
563       meas.getSecond().add(Arrays.getColumnAsDoubles(vals, i));
564     }
565     long[] timestamps = data.getTimeStamps();
566     for (Map.Entry<String, Pair<List<String>, List<double[]>>> entry : measurementsByUom.entrySet()) {
567       Pair<List<String>, List<double[]>> p = entry.getValue();
568       final List<String> measurementNames = p.getFirst();
569       final List<double[]> measurements = p.getSecond();
570       result.add(Charts.createTimeSeriesJFreeChart("chart for "
571               + info.getTableName() + ", sampleTime " + info.getSampleTime()
572               + " ms, generated by spf4j", timestamps,
573               measurementNames.toArray(new String[measurementNames.size()]), entry.getKey(),
574               measurements.toArray(new double[measurements.size()][])));
575     }
576     return result;
577   }

583   public void writeCsvTable(final String tableName, final File output) throws IOException {
584     TSTable table = getTSTable(tableName);
585     TimeSeries data = readAll(tableName);
586     DateTimeFormatter formatter = ISODateTimeFormat.dateTime();
587     try (Writer writer = new BufferedWriter(new OutputStreamWriter(
588             Files.newOutputStream(output.toPath()), Charsets.UTF_8))) {
589       Csv.writeCsvElement("timestamp", writer);
590       for (String colName : table.getColumnNames()) {
591         writer.append(',');
592         Csv.writeCsvElement(colName, writer);
593       }
594       writer.write('\n');
595       long[] timestamps = data.getTimeStamps();
596       long[][] values = data.getValues();
597       for (int i = 0; i < timestamps.length; i++) {
598         Csv.writeCsvElement(formatter.print(timestamps[i]), writer);
599         for (long val : values[i]) {
600           writer.append(',');
601           Csv.writeCsvElement(Long.toString(val), writer);
602         }
603         writer.write('\n');
604       }
605     }
606   }

608   public void writeCsvTables(final List<String> tableNames, final File output) throws IOException {
609     DateTimeFormatter formatter = ISODateTimeFormat.dateTime();
610     try (Writer writer = new BufferedWriter(new OutputStreamWriter(
611             Files.newOutputStream(output.toPath()), Charsets.UTF_8))) {
612       String firstTable = tableNames.get(0);
613       TSTable table = getTSTable(firstTable);
614       Csv.writeCsvElement("table", writer);
615       writer.append(',');
616       Csv.writeCsvElement("timestamp", writer);
617       for (String colName : table.getColumnNames()) {
618         writer.append(',');
619         Csv.writeCsvElement(colName, writer);
620       }
621       writer.write('\n');
622 
623       for (String tableName : tableNames) {
624         TimeSeries data = readAll(tableName);
625         long[] timestamps = data.getTimeStamps();
626         long[][] values = data.getValues();
627         for (int i = 0; i < timestamps.length; i++) {
628           Csv.writeCsvElement(tableName, writer);
629           writer.append(',');
630           Csv.writeCsvElement(formatter.print(timestamps[i]), writer);
631           for (long val : values[i]) {
632             writer.append(',');
633             Csv.writeCsvElement(Long.toString(val), writer);
634           }
635           writer.write('\n');
636         }
637       }
638     }

646   @SuppressFBWarnings("MDM_THREAD_YIELD")
647   public void tail(final long pollMillis, final long from, final TSDataHandler handler)
648           throws IOException {
649     Map<String, TSTable> lastState = new HashMap<>();
650     long lastSize = 0;
651     while (!Thread.interrupted() && !handler.finish()) {
652       long currSize = this.file.length();
653       if (currSize > lastSize) {
654         // see if we have new Tables;
655         reReadTableInfos();
656         Map<String, TSTable> currState = getTsTables();
657         for (String tableName : Sets.difference(currState.keySet(), lastState.keySet())) {
658           handler.newTable(tableName, currState.get(tableName).getColumnNames());
659         }
660         for (TSTable table : currState.values()) {
661           final String tableName = table.getTableName();
662           TSTable prevTableState = lastState.get(tableName);
663           final long currLastDataFragment = table.getLastDataFragment();
664 
665           if (prevTableState == null) {
666             long lastDataFragment = table.getFirstDataFragment();
667             if (lastDataFragment > 0) {
668               TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
669                       false);
670               handler.newData(tableName, data);
671             }
672           } else {
673             long lastDataFragment = prevTableState.getLastDataFragment();
674             if (lastDataFragment == 0) {
675               lastDataFragment = table.getFirstDataFragment();
676               if (lastDataFragment > 0) {
677                 TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
678                         false);
679                 handler.newData(tableName, data);
680               }
681             } else if (currLastDataFragment > lastDataFragment) {
682               TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
683                       true);
684               handler.newData(tableName, data);
685             }
686           }
687         }
688         lastState = currState;
689       }
690       lastSize = currSize;
691       try {
692         Thread.sleep(pollMillis);
693       } catch (InterruptedException ex) {
694         Thread.currentThread().interrupt();
695         break;
696       }
697     }
698   }

704   /**
705    * Quantized recorders will have min, max avg charts and distribution charts generated. Counting recorders will have
706    * simple charts generated.
707    *
708    * @param startTimeMillis
709    * @param endTimeMillis
710    * @param width
711    * @param height
712    * @return
713    * @throws IOException
714    */
715   @JmxExport(value = "generateChartsInterval",
716           description = "generate charts for all measurements in specified interval")
717   public List<String> generateCharts(final long startTimeMillis, final long endTimeMillis,
718           final int width, final int height) throws IOException {
719     try {
720       this.flush();
721       List<String> result = new ArrayList<>();
722       Collection<TSTable> columnsInfo = this.getTSTables();
723       for (TSTable info : columnsInfo) {
724         TimeSeries data = this.read(info.getTableName(), startTimeMillis, endTimeMillis);
725         if (data.getTimeStamps().length > 0) {
726           if (canGenerateMinMaxAvgCount(info)) {
727             result.add(generateMinMaxAvgCountChart(info, data, width, height));
728           }
729           if (canGenerateHeatChart(info)) {
730             result.add(generateHeatChart(info, data, width, height));
731           }
732         }
733       }
734       Multimap<String, TSTable> counters = getCounters(columnsInfo);
735       for (Map.Entry<String, Collection<TSTable>> entry : counters.asMap().entrySet()) {
736         Collection<TSTable> ltables = entry.getValue();
737         int l = ltables.size();
738         long[][] timestamps = new long[l][];
739         double[][] cdata = new double[l][];
740         double[][] cdata2 = new double[l][];
741         int i = 0;
742         String[] measurementNames = new String[cdata.length];
743         String[] measurementNames2 = new String[cdata2.length];
744         String uom1 = "count";
745         String uom2 = "";
746         for (TSTable colInfo : ltables) {
747           TimeSeries data = this.read(colInfo.getTableName(), startTimeMillis, endTimeMillis);
748           timestamps[i] = data.getTimeStamps();
749           final long[][] values = data.getValues();
750           cdata[i] = Arrays.getColumnAsDoubles(values, colInfo.getColumnIndex("count"));
751           cdata2[i] = Arrays.getColumnAsDoubles(values, colInfo.getColumnIndex("total"));
752           measurementNames[i] = colInfo.getTableName() + ".count";
753           measurementNames2[i] = colInfo.getTableName() + ".total";
754           uom2 = new String(colInfo.getTableMetaData(), Charsets.UTF_8);
755           i++;
756         }
757         result.add(generateCountChart(entry.getKey(), timestamps, measurementNames,
758                 measurementNames2, uom1, uom2, cdata, cdata2, width, height));
759       }
760       return result;
761     } catch (IOException | RuntimeException ex) {
762       throw ex;
763     }
764   }

766   private static Multimap<String, TSTable> getCounters(final Collection<TSTable> columnInfos) {
767     Multimap<String, TSTable> result = HashMultimap.create();
768     for (TSTable info : columnInfos) {
769       if (isCounterOnly(info)) {
770         String groupName = info.getTableName();
771         if (groupName.startsWith("(")) {
772           int cidx = groupName.indexOf(',');
773           if (cidx > 0) {
774             groupName = groupName.substring(1, cidx);
775           }
776         }
777         result.put(groupName, info);
778       }
779     }
780     return result;
781   }

810   private String generateMinMaxAvgCountChart(
811           final TSTable info, final TimeSeries data,
812           final int width, final int height) throws IOException {
813     long[][] vals = data.getValues();
814     double[] min = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("min"));
815     double[] max = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("max"));
816     double[] total = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("total"));
817     double[] count = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("count"));
818     for (int i = 0; i < count.length; i++) {
819       if (count[i] == 0) {
820         min[i] = 0;
821         max[i] = 0;
822       }
823     }
824     long[] timestamps = data.getTimeStamps();
825     BufferedImage combined = Charts.createMinMaxAvgCountImg("Measurements for "
826             + info.getTableName() + " generated by spf4j",
827             timestamps, min, max, total, count, new String(info.getTableMetaData(), Charsets.UTF_8), width, height);
828     File dbFile = new File(this.getDBFilePath());
829     File graphicFile = File.createTempFile(dbFile.getName() + '_' + fixName(info.getTableName()), ".mmac.png",
830             dbFile.getParentFile());
831     ImageIO.write(combined, "png", graphicFile);
832     return graphicFile.getPath();
833   }

862   private static String fixName(final String name) {
863     final int length = name.length();
864     StringBuilder result = new StringBuilder(length);
865     for (int i = 0; i < length; i++) {
866       char c = name.charAt(i);
867       if (Character.isJavaIdentifierPart(c)) {
868         result.append(c);
869       }
870     }
871     return result.toString();
872   }
